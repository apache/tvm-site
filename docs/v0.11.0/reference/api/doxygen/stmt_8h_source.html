789  template <typename... Args>

791  Array<Stmt> seq;

802  if (!stmt.defined()) return;
803  if (auto* op = stmt.as<SeqStmtNode>()) {
804  operator()(0, op->seq);
805  } else {
806  seq_->push_back(stmt);
807  }
808  }

812  for (auto v : seq) {

814  }
