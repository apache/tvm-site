761 // common rule for RetValue and ArgValue
762 template <>
763 struct PackedFuncValueConverter<PrimExpr> {
764  static PrimExpr From(const TVMPODValue_& val) {
765  if (val.type_code() == kTVMNullptr) {
766  return PrimExpr(ObjectPtr<Object>(nullptr));
767  }
768  if (val.type_code() == kDLInt) {
769  int64_t value = val.operator int64_t();
770  if (value > std::numeric_limits<int>::max() || value < std::numeric_limits<int>::min()) {
771  return IntImm(runtime::DataType::Int(64), value);
772  }
773  return IntImm(runtime::DataType::Int(32), val.operator int());
774  }
775  if (val.type_code() == kDLFloat) {
776  return FloatImm(runtime::DataType::Float(32), val.operator double());
777  }
778 
779  return PrimExpr::FromObject_(val.AsObjectRef<ObjectRef>());
780  }
781 };
