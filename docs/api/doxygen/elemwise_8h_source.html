79 inline Tensor fast_tanh_float(const Tensor& in, std::string name, std::string tag) {
80  // Clamp the inputs to the range [-9, 9] since anything outside
81  // this range is +/-1.0f in single-precision.
82  auto x = maximum(minimum(in, make_const(in->dtype, 9.0)), make_const(in->dtype, -9.0));
83 
84  // The monomial coefficients of the numerator polynomial (odd).
85  auto alpha_1 = make_const(in->dtype, 4.89352455891786e-03);
86  auto alpha_3 = make_const(in->dtype, 6.37261928875436e-04);
87  auto alpha_5 = make_const(in->dtype, 1.48572235717979e-05);
88  auto alpha_7 = make_const(in->dtype, 5.12229709037114e-08);
89  auto alpha_9 = make_const(in->dtype, -8.60467152213735e-11);
90  auto alpha_11 = make_const(in->dtype, 2.00018790482477e-13);
91  auto alpha_13 = make_const(in->dtype, -2.76076847742355e-16);
92 
93  // The monomial coefficients of the denominator polynomial (even).
94  auto beta_0 = make_const(in->dtype, 4.89352518554385e-03);
95  auto beta_2 = make_const(in->dtype, 2.26843463243900e-03);
96  auto beta_4 = make_const(in->dtype, 1.18534705686654e-04);
97  auto beta_6 = make_const(in->dtype, 1.19825839466702e-06);
98 
99  return compute(
100  x->shape,
101  [&](const Array<Var>& i) {
102  auto x2 = x(i) * x(i);
103  auto p = x2 * alpha_13 + alpha_11;
104  p = x2 * p + alpha_9;
105  p = x2 * p + alpha_7;
106  p = x2 * p + alpha_5;
107  p = x2 * p + alpha_3;
108  p = x2 * p + alpha_1;
109  p = x(i) * p;
110 
111  auto q = x2 * beta_6 + beta_4;
112  q = x2 * q + beta_2;
113  q = x2 * q + beta_0;
114  return p / q;
115  },
116  name, tag);
117 }

402 inline Tensor fast_exp_float32(const Tensor& _x, std::string name, std::string tag) {
403  auto x_hi = make_const(DataType::Float(32), 88.3762626647950f);
404  auto x_lo = make_const(DataType::Float(32), -88.3762626647949f);
405  auto log2e = make_const(DataType::Float(32), 1.44269504088896341f);
406  auto ln2 = make_const(DataType::Float(32), 0.6931471805599453f);
407  PrimExpr p[6] = {make_const(DataType::Float(32), 1.9875691500E-4f),
408  make_const(DataType::Float(32), 1.3981999507E-3f),
409  make_const(DataType::Float(32), 8.3334519073E-3f),
410  make_const(DataType::Float(32), 4.1665795894E-2f),
411  make_const(DataType::Float(32), 1.6666665459E-1f),
412  make_const(DataType::Float(32), 5.0000001201E-1f)};
413  auto one = make_const(DataType::Float(32), 1.0f);
414  auto one_half = make_const(DataType::Float(32), 0.5f);
415  auto b = make_const(DataType::Float(32), 127.0f);
416 
417  return compute(
418  _x->shape,
419  [&](const Array<Var>& i) {
420  // clamp x
421  auto x = ::tvm::max(::tvm::min(_x(i), x_hi), x_lo);
422  // integer part
423  auto n = ::tvm::floor(x * log2e + one_half);
424  // fractional part
425  auto f = x - n * ln2;
426  auto y =
427  (((((p[0] * f + p[1]) * f + p[2]) * f + p[3]) * f + p[4]) * f + p[5]) * f * f + f + one;
428  // Return 2^m * exp(r).
429  auto ef =
430  tvm::reinterpret(DataType::Float(32), ::tvm::cast(DataType::Int(32), n + b) << 23);
431  return ::tvm::max(ef * y, _x(i)); // NOLINT(*)
432  },
433  name, tag);
434 }

461 inline Tensor fast_erf_float32(const Tensor& data, std::string name, std::string tag) {
462  auto plus_4 = make_const(DataType::Float(32), 4.f);
463  auto minus_4 = make_const(DataType::Float(32), -4.f);
464 
465  // The monomial coefficients of the numerator polynomial (odd).
466  auto alpha_1 = make_const(DataType::Float(32), -1.60960333262415e-02f);
467  auto alpha_3 = make_const(DataType::Float(32), -2.95459980854025e-03f);
468  auto alpha_5 = make_const(DataType::Float(32), -7.34990630326855e-04f);
469  auto alpha_7 = make_const(DataType::Float(32), -5.69250639462346e-05f);
470  auto alpha_9 = make_const(DataType::Float(32), -2.10102402082508e-06f);
471  auto alpha_11 = make_const(DataType::Float(32), 2.77068142495902e-08f);
472  auto alpha_13 = make_const(DataType::Float(32), -2.72614225801306e-10f);
473 
474  // The monomial coefficients of the denominator polynomial (even).
475  auto beta_0 = make_const(DataType::Float(32), -1.42647390514189e-02f);
476  auto beta_2 = make_const(DataType::Float(32), -7.37332916720468e-03f);
477  auto beta_4 = make_const(DataType::Float(32), -1.68282697438203e-03f);
478  auto beta_6 = make_const(DataType::Float(32), -2.13374055278905e-04f);
479  auto beta_8 = make_const(DataType::Float(32), -1.45660718464996e-05f);
480 
481  return compute(
482  data->shape,
483  [&](const Array<Var>& i) {
484  // clamp x
485  auto x = tvm::max(tvm::min(data(i), plus_4), minus_4);
486  auto x2 = x * x;
487 
488  // Evaluate the numerator polynomial p.
489  auto p = x2 * alpha_13 + alpha_11;
490  p = x2 * p + alpha_9;
491  p = x2 * p + alpha_7;
492  p = x2 * p + alpha_5;
493  p = x2 * p + alpha_3;
494  p = x2 * p + alpha_1;
495  p = x * p;
496 
497  // Evaluate the denominator polynomial p.
498  auto q = x2 * beta_8 + beta_6;
499  q = x2 * q + beta_4;
500  q = x2 * q + beta_2;
501  q = x2 * q + beta_0;
502 
503  return p / q;
504  },
505  name, tag);
506 }
