64 inline PrimExpr bilinear_sample_nchw(const Tensor& input, const Array<PrimExpr>& indices,
65  const PrimExpr max_y, const PrimExpr max_x) {
66  auto in_y = indices[2];
67  auto yf = tvm::floor(in_y);
68  auto yc = tvm::cast(DataType::Int(32), tvm::ceil(in_y));
69 
70  auto y0 = tvm::cast(DataType::Int(32), tvm::floor(in_y));
71  auto y1 = tvm::if_then_else((yc > max_y), max_y, yc);
72  auto y_lerp = in_y - yf;
73 
74  auto in_x = indices[3];
75  auto xf = tvm::floor(in_x);
76  auto xc = tvm::cast(DataType::Int(32), tvm::ceil(in_x));
77 
78  auto x0 = tvm::cast(DataType::Int(32), tvm::floor(in_x));
79  auto x1 = tvm::if_then_else((xc > max_x), max_x, xc);
80  auto x_lerp = in_x - xf;
81 
82  auto A = input(indices[0], indices[1], y0, x0);
83  auto B = input(indices[0], indices[1], y0, x1);
84  auto C = input(indices[0], indices[1], y1, x0);
85  auto D = input(indices[0], indices[1], y1, x1);

87  return A * (1 - x_lerp) * (1 - y_lerp) + B * x_lerp * (1 - y_lerp) + C * (1 - x_lerp) * y_lerp +

89 }
