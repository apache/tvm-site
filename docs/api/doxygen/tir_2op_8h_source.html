723 template <typename ValueType>
724 inline PrimExpr MakeConstScalar(DataType t, ValueType value) {
725  if (t.is_int()) return IntImm(t, static_cast<int64_t>(value));
726  if (t.is_uint()) {
727  // Use IntImm if it is a small integer
728  uint64_t uval = static_cast<uint64_t>(value);
729  if (uval <= static_cast<uint64_t>(std::numeric_limits<int64_t>::max())) {
730  return IntImm(t, static_cast<int64_t>(value));
731  } else {
732  uint64_t mask = (static_cast<uint64_t>(1) << 32U) - 1U;
733  uint64_t low = uval & mask;
734  uint64_t high = uval >> 32U;
735  return LargeUIntImm(t, static_cast<int64_t>(low), static_cast<int64_t>(high));
736  }
737  }
738  if (t.is_float()) return FloatImm(t, static_cast<double>(value));
739  // For now, we store const scalar values of custom datatypes within doubles; later, during the
740  // datatypes lowering pass, we will lower the value to its true representation in the format
741  // specified by the datatype.
742  // TODO(gus) when do we need to start worrying about doubles not being precise enough?
743  if (static_cast<uint8_t>(t.code()) >= static_cast<uint8_t>(kTVMCustomBegin)) {
744  return FloatImm(t, static_cast<double>(value));
745  }
746  LOG(FATAL) << "cannot make const for type " << t;
747  return PrimExpr();
748 }
