184 inline TFunc WithAttr(TFunc func, const std::string& attr_key, ObjectRef attr_value) {
185  using TNode = typename TFunc::ContainerType;
186  static_assert(TNode::_type_final, "Can only operate on the leaf nodes");
187  TNode* node = func.CopyOnWrite();
188  if (node->attrs.defined()) {

190  } else {
191  Map<String, ObjectRef> dict = {{attr_key, attr_value}};
192  node->attrs = DictAttrs(dict);
193  }
194  return func;
195 }
