184 template <typename TFunc,
185  typename = typename std::enable_if<std::is_base_of<BaseFunc, TFunc>::value>::type>
186 inline TFunc WithAttr(TFunc func, const std::string& attr_key, ObjectRef attr_value) {
187  using TNode = typename TFunc::ContainerType;
188  static_assert(TNode::_type_final, "Can only operate on the leaf nodes");
189  TNode* node = func.CopyOnWrite();
190  if (node->attrs.defined()) {
191  node->attrs.CopyOnWrite()->dict.Set(attr_key, attr_value);
192  } else {
193  Map<String, ObjectRef> dict = {{attr_key, attr_value}};
194  node->attrs = DictAttrs(dict);
195  }
196  return func;
197 }
