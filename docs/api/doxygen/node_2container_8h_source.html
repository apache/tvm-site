668  bool TryInsert(const key_type& key, ListNode* result) {
669  if (slots_ == 0) {
670  return false;
671  }
672  // required that `iter` to be the head of a linked list through which we can iterator
673  ListNode iter = IndexFromHash(ObjectHash()(key));
674  // `iter` can be: 1) empty; 2) body of an irrelevant list; 3) head of the relevant list
675  // Case 1: empty
676  if (iter.IsEmpty()) {
677  iter.NewHead(KVType(key, ObjectRef(nullptr)));
678  this->size_ += 1;
679  *result = iter;
680  return true;
681  }
682  // Case 2: body of an irrelevant list
683  if (!iter.IsHead()) {
684  // we move the elements around and construct the single-element linked list
685  return IsFull() ? false : TrySpareListHead(iter, key, result);
686  }
687  // Case 3: head of the relevant list
688  // we iterate through the linked list until the end
689  // make sure `iter` is the previous element of `next`
690  ListNode next = iter;
691  do {
692  // find equal item, do not insert
693  if (ObjectEqual()(key, next.Key())) {
694  *result = next;
695  return true;
696  }
697  // make sure `iter` is the previous element of `next`
698  iter = next;
699  } while (next.MoveToNext(this));
700  // `iter` is the tail of the linked list
701  // always check capacity before insertion
702  if (IsFull()) {
703  return false;
704  }
705  // find the next empty slot
706  uint8_t jump;
707  if (!iter.GetNextEmpty(this, &jump, result)) {
708  return false;
709  }
710  result->NewTail(KVType(key, ObjectRef(nullptr)));
711  // link `iter` to `empty`, and move forward
712  iter.SetJump(jump);
713  this->size_ += 1;
714  return true;
715  }
727  bool TrySpareListHead(ListNode target, const key_type& key, ListNode* result) {
728  // `target` is not the head of the linked list
729  // move the original item of `target` (if any)
730  // and construct new item on the position `target`
731  // To make `target` empty, we
732  // 1) find `w` the previous element of `target` in the linked list
733  // 2) copy the linked list starting from `r = target`
734  // 3) paste them after `w`
735  // read from the linked list after `r`
736  ListNode r = target;
737  // write to the tail of `w`
738  ListNode w = target.FindPrev(this);
739  // after `target` is moved, we disallow writing to the slot
740  bool is_first = true;
741  uint8_t r_meta, jump;
742  ListNode empty;
743  do {
744  // `jump` describes how `w` is jumped to `empty`
745  // rehash if there is no empty space after `w`
746  if (!w.GetNextEmpty(this, &jump, &empty)) {
747  return false;
748  }
749  // move `r` to `empty`
750  empty.NewTail(std::move(r.Data()));
751  // clear the metadata of `r`
752  r_meta = r.Meta();
753  if (is_first) {
754  is_first = false;
755  r.SetProtected();
756  } else {
757  r.SetEmpty();
758  }
759  // link `w` to `empty`, and move forward
760  w.SetJump(jump);
761  w = empty;
762  // move `r` forward as well
763  } while (r.MoveToNext(this, r_meta));
764  // finally we have done moving the linked list
765  // fill data_ into `target`
766  target.NewHead(KVType(key, ObjectRef(nullptr)));
767  this->size_ += 1;
768  *result = target;
769  return true;
770  }
775  void Erase(const ListNode& iter) {
776  this->size_ -= 1;
777  if (!iter.HasNext()) {
778  // `iter` is the last
779  if (!iter.IsHead()) {

789  iter.Data() = std::move(last.Data());

791  prev.SetJump(0);
792  }

959  static void CalcTableSize(uint64_t cap, uint32_t* fib_shift, uint64_t* n_slots) {
960  uint32_t shift = 64;
961  uint64_t slots = 1;
962  for (uint64_t c = cap; c; c >>= 1) {
963  shift -= 1;
964  slots <<= 1;
965  }
966  ICHECK_GT(slots, cap);
