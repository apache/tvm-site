68 inline Tensor dilate(const Tensor& x, Array<PrimExpr> strides, std::string name = "tensor",
69  std::string tag = kInjective) {
70  auto n = x->shape.size();
71  CHECK_EQ(n, strides.size()) << "strides size (" << strides.size()
72  << ") must match dimension of x (" << n << ")";
73 
74  Array<PrimExpr> out_shape;
75  arith::Analyzer analyzer;
76  for (size_t i = 0; i < n; ++i) {
77  out_shape.push_back(
78  analyzer.Simplify((x->shape[i] - 1) * cast(DataType::Int(32), strides[i] + 1)));
79  }
80 
81  return tvm::te::compute(
82  out_shape,
83  [&](const Array<Var>& indices) {
84  Array<PrimExpr> not_zero;
85  Array<PrimExpr> index_tuple;
86  for (size_t i = 0; i < n; ++i) {
87  if (IsConstInt(strides[i]) && GetConstInt(strides[i]) == 1) {
88  index_tuple.push_back(indices[i]);
89  } else {
90  index_tuple.push_back(indexdiv(indices[i], strides[i]));
91  not_zero.push_back((indexmod(indices[i], strides[i])) == 0);
92  }
93  }
94  if (not_zero.size() > 0) {
95  auto all_not_zero = all(not_zero);
96  return tvm::if_then_else(all_not_zero, x(index_tuple), make_const(x->dtype, 0));
97  }
98  return x(index_tuple);
99  },
100  name, tag);
101 }
