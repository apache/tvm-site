42 inline Schedule schedule_lrn(const Array<Tensor>& outs) {
43  Array<Operation> out_ops;
44  for (auto t : outs) {
45  out_ops.push_back(t->op);
46  }
47  Schedule s = create_schedule(out_ops);
48  int num_thread = 64;
49  IterVar block_x = tvm::te::thread_axis(Range(), "blockIdx.x");
50  IterVar thread_x = tvm::te::thread_axis(Range(0, num_thread), "threadIdx.x");
51  Tensor lrn = outs[0];
52  Tensor sqr_sum_up = lrn->op->InputTensors()[1];
53  Tensor sqr_sum = sqr_sum_up->op->InputTensors()[0];
54  Tensor set_pad = sqr_sum->op->InputTensors()[0];
55  s[set_pad].bind(set_pad->op.as<ComputeOpNode>()->axis[0], block_x);
56  IterVar rxk = sqr_sum->op.as<ComputeOpNode>()->reduce_axis[0];
57  IterVar xko, xki;
58  s[sqr_sum].split(rxk, num_thread, &xko, &xki);
59  Tensor srf = s.rfactor(sqr_sum, xki)[0];
60  s[sqr_sum].bind(s[sqr_sum]->op.as<ComputeOpNode>()->axis[0], block_x);
61  s[sqr_sum].bind(s[sqr_sum]->op.as<ComputeOpNode>()->reduce_axis[0], thread_x);
62  s[srf].compute_at(s[sqr_sum], s[sqr_sum]->op.as<ComputeOpNode>()->reduce_axis[0]);
63  s[sqr_sum_up].bind(sqr_sum_up->op.as<ComputeOpNode>()->axis[0], block_x);
64  IterVar xto, xti;
65  s[lrn].split_by_nparts(lrn->op.as<ComputeOpNode>()->axis[1], num_thread, &xto, &xti);
66  s[lrn].bind(lrn->op.as<ComputeOpNode>()->axis[0], block_x);
67  s[lrn].bind(xto, thread_x);
68 
69  return s;
70 }
