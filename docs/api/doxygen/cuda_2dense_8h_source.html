52 inline tvm::te::Tensor dense_cuda(const Target& target, const tvm::te::Tensor& data,
53  const tvm::te::Tensor& weight, const tvm::te::Tensor& bias,
54  const DataType& out_dtype) {
55  CHECK_EQ(data->shape.size(), 2) << "dense requires 2-D data";
56  CHECK_EQ(weight->shape.size(), 2) << "dense requires 2-D weight";
57  if (bias.defined()) {
58  CHECK_EQ(bias->shape.size(), 1) << "dense requires 1-D bias";
59  }
60 
61  auto batch = data->shape[0];
62  auto in_dim = data->shape[1];
63  auto out_dim = weight->shape[0];
64 
65  if (target->GetLibs().count("cublas")) {
66  CHECK_EQ(data->dtype, out_dtype) << "Mixed precision not supported.";
67  auto mm = topi::contrib::cublas_matmul(data, weight, false, true);
68  if (bias.defined()) {
69  mm = tvm::te::compute(
70  {batch, out_dim}, [&](Var i, Var j) { return mm(i, j) + bias(j); }, "tensor", kBroadcast);
71  }
72 
73  return mm;
74  } else {
75  return topi::nn::dense(data, weight, bias, out_dtype);
76  }
77 }

87 inline Schedule schedule_dense(const Target& target, const Array<Tensor>& outs) {
88  if (target->id->name == "cuda" && target->GetLibs().count("cublas")) {
89  return topi::generic::schedule_extern(target, outs);
90  }
91 
92  Array<Operation> out_ops;
93  for (auto t : outs) {
94  out_ops.push_back(t->op);
95  }
96  auto s = create_schedule(out_ops);
97 
98  auto _schedule = [&](const Tensor& dense) {
99  auto num_thread = 64;
100  auto k = dense->op.as<ComputeOpNode>()->reduce_axis[0];
101  IterVar ko, kf;
102  s[dense].split(k, num_thread, &ko, &kf);
103  auto dense_f = s.rfactor(dense, kf)[0];
104 
105  Tensor out;
106  if (detail::contains(s->outputs, dense->op)) {
107  out = dense;
108  } else {
109  out = outs[0]->op.output(0);
110  s[dense].compute_at(s[out], s[out]->op.as<ComputeOpNode>()->axis[1]);
111  }
112  s[out].bind(s[out]->op.as<ComputeOpNode>()->axis[0],
113  tvm::te::thread_axis(Range(), "blockIdx.y"));
114  s[out].bind(s[out]->op.as<ComputeOpNode>()->axis[1],
115  tvm::te::thread_axis(Range(), "blockIdx.x"));
116 
117  auto tx = s[dense]->op.as<ComputeOpNode>()->reduce_axis[0];
118  auto thread_x = tvm::te::thread_axis(Range(), "threadIdx.x");
119  s[dense].bind(tx, thread_x);
120  s[dense_f].compute_at(s[dense], tx);
121  s[dense].set_store_predicate(static_cast<PrimExpr>(thread_x) == 0);
122  s[out].set_store_predicate(static_cast<PrimExpr>(thread_x) == 0);
123  };
124 
125  std::function<void(Operation)> traverse;
126  traverse = [&](const Operation& op) {
127  // Inline all one-to-one-mapping operators except the last stage (output)
128  if (is_broadcast(op->tag)) {
129  if (!detail::contains(s->outputs, op)) {
130  s[op].compute_inline();
131  }
132  for (auto tensor : op->InputTensors()) {
133  if (tensor->op->InputTensors().size() > 0) {
134  traverse(tensor->op);
135  }
136  }
137  } else if (op->tag == "dense") {
138  // If tag starts with global_pool
139  auto dense = op.output(0);
140  _schedule(dense);
141  } else {
142  LOG(ERROR) << "Unsupported operator " << op->tag;
143  }
144  };
145 
146  traverse(outs[0]->op);
147  return s;
148 }
