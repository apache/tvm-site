528 struct AttrInitEntry {
529  // The attributes
530  using TSelf = AttrInitEntry<T>;
531  // The type key
532  const char* type_key_;
533  // field name
534  const char* key_;
535  // internal value.
536  T* value_;
537  // whether the value is missing.
538  // NOTE: initialize to false so that the destructor does not throw unless
539  // AttrInitVisitor::operator() is committed to returning an instance of this class.
540  // It is expected not to set this to true until that is true.
541  bool value_missing_{false};
542 
543  AttrInitEntry() = default;
544 
545  AttrInitEntry(AttrInitEntry&& other) {
546  type_key_ = other.type_key_;
547  key_ = other.key_;
548  value_ = other.value_;
549  value_missing_ = other.value_missing_;
550  // avoid unexpected throw
551  other.value_missing_ = false;
552  }
553 
554  // If the value is still missing in destruction time throw an error.
555  ~AttrInitEntry() DMLC_THROW_EXCEPTION {
556  if (value_missing_) {
557  std::ostringstream os;
558  os << type_key_ << ": Cannot find required field \'" << key_ << "\' during initialization. "
559  << "If the key is defined check that its type matches the declared type.";
560  throw AttrError(os.str());
561  }
562  }
563  // override fields.
564  // This function sets the lower bound of the attribute
565  TSelf& set_lower_bound(const T& begin) {
566  if (this->value_missing_) return *this;
567  const T& val = *value_;
568  if (begin > val) {
569  std::ostringstream os;
570  os << type_key_ << "." << key_ << "'s "
571  << "value " << val << " is smaller than the lower bound " << begin;
572  throw AttrError(os.str());
573  }
574  return *this;
575  }
576  // This function sets the upper bound of the attribute
577  TSelf& set_upper_bound(const T& end) {
578  if (this->value_missing_) return *this;
579  const T& val = *value_;
580  if (val > end) {
581  std::ostringstream os;
582  os << type_key_ << "." << key_ << "'s "
583  << "value " << val << " is bigger than the upper bound " << end;
584  throw AttrError(os.str());
585  }
586  return *this;
587  }
588  // set default when
589  TSelf& set_default(const T& value) {
590  if (!value_missing_) return *this;
591  *value_ = value;
592  value_missing_ = false;
593  return *this;
594  }
595  TSelf& describe(DMLC_ATTRIBUTE_UNUSED const char* str) { return *this; }

881  void InitByPackedArgs(const runtime::TVMArgs& args, bool allow_unknown) final {
882  ICHECK_EQ(args.size() % 2, 0);
883  const int kLinearSearchBound = 16;
884  int hit_count = 0;
885  // applies two strategies to lookup
886  if (args.size() < kLinearSearchBound) {
887  // linear search.
888  auto ffind = [&args](const char* key, ffi::AnyView* val) {
889  for (int i = 0; i < args.size(); i += 2) {
890  if (!std::strcmp(key, args[i].cast<const char*>())) {
891  *val = args[i + 1];
892  return true;
893  }
894  }
895  return false;
896  };
897  auto vis = ::tvm::detail::CreateInitVisitor(DerivedType::_type_key, ffind);
898  self()->_tvm_VisitAttrs(vis);
899  hit_count = vis.hit_count_;
900  } else {
901  // construct a map then do lookup.
902  std::unordered_map<std::string, runtime::TVMArgValue> kwargs;
903  for (int i = 0; i < args.size(); i += 2) {
904  kwargs[args[i].cast<std::string>()] = args[i + 1];
905  }
906  auto ffind = [&kwargs](const char* key, ffi::AnyView* val) {
907  auto it = kwargs.find(key);
908  if (it != kwargs.end()) {
909  *val = it->second;
910  return true;
911  }
912  return false;
913  };
914  auto vis = ::tvm::detail::CreateInitVisitor(DerivedType::_type_key, ffind);
915  self()->_tvm_VisitAttrs(vis);
916  hit_count = vis.hit_count_;
917  }
918  // error handling, slow path
919  if (hit_count * 2 != args.size() && !allow_unknown) {

927  << "\', Possible fields:\n";
928  os << "----------------\n";
929  this->PrintDocString(os);
930  throw AttrError(os.str());
931  }
932  }
933  }
934  }
