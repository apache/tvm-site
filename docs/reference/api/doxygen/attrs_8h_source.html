246 inline TFunc WithAttr(TFunc input, const std::string& attr_key, Any attr_value) {
247  using TNode = typename TFunc::ContainerType;
248  static_assert(TNode::_type_final, "Can only operate on the leaf nodes");
249  TNode* node = input.CopyOnWrite();

267 template <typename TFunc>
268 inline TFunc WithAttrs(TFunc input, ffi::Map<ffi::String, Any> attrs) {
269  using TNode = typename TFunc::ContainerType;
270  static_assert(TNode::_type_final, "Can only operate on the leaf nodes");
271  if (attrs.empty()) return input;
272  TNode* node = input.CopyOnWrite();
273  // node->attrs is NOTNULLABLE by contract, but defend against a caller
274  // that left a moved-from DictAttrs in place by re-initializing here.
275  if (!node->attrs.defined()) node->attrs = DictAttrs();
276  auto* dict_node = node->attrs.CopyOnWrite();
277  for (const auto& [k, v] : attrs) {
278  dict_node->dict.Set(k, v);
279  }
280  return input;
