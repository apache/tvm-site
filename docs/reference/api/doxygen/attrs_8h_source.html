159  template <typename TObjectRef>
160  ffi::Optional<TObjectRef> GetAttr(
161  const std::string& attr_key,
162  ffi::Optional<TObjectRef> default_value = ffi::Optional<TObjectRef>(std::nullopt)) const {
163  if (!defined()) return default_value;
164  const DictAttrsNode* node = this->as<DictAttrsNode>();
165  auto it = node->dict.find(attr_key);
166  if (it != node->dict.end()) {
167  return (*it).second.cast<TObjectRef>();
168  } else {
169  return default_value;
170  }
171  }
