179  // Utils for accessing attributes
180  // This needs to be on DictAttrs, not DictAttrsNode because we return the default
181  // value if DictAttrsNode is not defined.
201  template <typename TObjectRef>
202  Optional<TObjectRef> GetAttr(
203  const std::string& attr_key,
204  Optional<TObjectRef> default_value = Optional<TObjectRef>(std::nullopt)) const {
205  if (!defined()) return default_value;
206  const DictAttrsNode* node = this->as<DictAttrsNode>();
207  auto it = node->dict.find(attr_key);
208  if (it != node->dict.end()) {
209  return (*it).second.cast<TObjectRef>();
210  } else {
211  return default_value;
212  }
213  }
214  // variant that uses TObjectRef to enable implicit conversion to default value.
215  template <typename TObjectRef>
216  Optional<TObjectRef> GetAttr(const std::string& attr_key, TObjectRef default_value) const {
217  return GetAttr<TObjectRef>(attr_key, Optional<TObjectRef>(default_value));
218  }
238  bool HasNonzeroAttr(const std::string& attr_key) const {
239  return GetAttr<Integer>(attr_key, 0).value_or(0).IntValue() != 0;

389 class AttrsNodeReflAdapter : public BaseAttrsNode {
390  public:
391  void InitByPackedArgs(const ffi::PackedArgs& args, bool allow_unknown) final {
392  LOG(FATAL) << "`" << DerivedType::_type_key << "` uses new reflection mechanism for init";

395  bool SEqualReduce(const DerivedType* other, SEqualReducer equal) const {
396  const TVMFFITypeInfo* type_info = TVMFFIGetTypeInfo(DerivedType::RuntimeTypeIndex());
397  bool success = true;
398  ffi::reflection::ForEachFieldInfoWithEarlyStop(
399  type_info, [&](const TVMFFIFieldInfo* field_info) {
400  ffi::reflection::FieldGetter field_getter(field_info);
401  ffi::Any field_value = field_getter(self());
402  ffi::Any other_field_value = field_getter(other);
403  if (!equal.AnyEqual(field_value, other_field_value)) {
404  success = false;
405  return true;
406  }
407  return false;
408  });
409  return success;
410  }
411 
412  void SHashReduce(SHashReducer hash_reducer) const {
413  const TVMFFITypeInfo* type_info = TVMFFIGetTypeInfo(DerivedType::RuntimeTypeIndex());
414  ffi::reflection::ForEachFieldInfo(type_info, [&](const TVMFFIFieldInfo* field_info) {
415  ffi::reflection::FieldGetter field_getter(field_info);
416  ffi::Any field_value = field_getter(self());
417  hash_reducer(field_value);
418  });
419  }
420 
421  private:
422  DerivedType* self() const {
423  return const_cast<DerivedType*>(static_cast<const DerivedType*>(this));
424  }

432 template <typename TAttrs>
433 inline TAttrs AttrsWithDefaultValues() {
434  static_assert(std::is_base_of_v<Attrs, TAttrs>, "Can only take attr nodes");
435  using ContainerType = typename TAttrs::ContainerType;
436  if constexpr (std::is_base_of_v<AttrsNodeReflAdapter<ContainerType>, ContainerType>) {
437  static auto finit_object = ffi::Function::GetGlobalRequired("ffi.MakeObjectFromPackedArgs");
438  AnyView packed_args[1];
439  packed_args[0] = ContainerType::RuntimeTypeIndex();
440  ffi::Any rv;
441  finit_object.CallPacked(ffi::PackedArgs(packed_args, 1), &rv);
442  return rv.cast<TAttrs>();
443  } else {
444  auto n = make_object<ContainerType>();
