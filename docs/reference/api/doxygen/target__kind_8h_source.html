286 template <typename ValueType>
287 struct ValueTypeInfoMaker<ValueType, std::false_type, std::false_type> {
288  using ValueTypeInfo = TargetKindNode::ValueTypeInfo;
289 
290  ValueTypeInfo operator()() const {
291  ValueTypeInfo info;
292  info.key = nullptr;
293  info.val = nullptr;
294  if constexpr (std::is_base_of_v<ObjectRef, ValueType>) {
295  int32_t tindex = ffi::TypeToRuntimeTypeIndex<ValueType>::v();
296  info.type_index = tindex;
297  info.type_key = runtime::Object::TypeIndex2Key(tindex);
298  return info;
299  } else if constexpr (std::is_same_v<ValueType, ffi::String>) {
300  // special handle string since it can be backed by multiple types.
301  info.type_index = ffi::TypeIndex::kTVMFFIStr;
302  info.type_key = ffi::TypeTraits<ValueType>::TypeStr();
303  return info;
304  } else {
305  // TODO(tqchen) consider upgrade to leverage any system to support union type
306  constexpr int32_t tindex = ffi::TypeToFieldStaticTypeIndex<ValueType>::value;
307  static_assert(tindex != ffi::TypeIndex::kTVMFFIAny, "Do not support union type for now");
308  info.type_index = tindex;
309  info.type_key = runtime::Object::TypeIndex2Key(tindex);
310  return info;
311  }
312  }
313 };
