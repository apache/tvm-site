366 inline ffi::Optional<T> MatchStructInfo(const Expr& expr) {
367  using TNode = typename T::ContainerType;
368  if (const TNode* ptr = expr->struct_info_.as<TNode>()) {
369  return ffi::GetRef<T>(ptr);
370  } else {
371  return std::nullopt;
372  }
