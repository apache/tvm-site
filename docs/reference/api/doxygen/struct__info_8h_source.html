416 inline Optional<T> MatchStructInfo(const Expr& expr) {
417  using TNode = typename T::ContainerType;
418  if (const TNode* ptr = expr->struct_info_.as<TNode>()) {
419  return GetRef<T>(ptr);

421  return NullOpt;
