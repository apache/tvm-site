866  inline PrimExpr OpName(PrimExpr x, Span span = Span()) { \
867  static const Op& op = Op::Get("tir." #OpName); \
868  if (x.dtype().is_bfloat16()) { \
869  DataType srcType = x.dtype(); \
870  DataType dstType(kDLFloat, 32, srcType.lanes()); \
871  PrimExpr castX = tir::Cast(dstType, {x}, span); \
872  PrimExpr result = tir::Call(dstType, op, {castX}, span); \
873  return tir::Cast(srcType, {result}, span); \
874  } else { \
875  return tir::Call(x.dtype(), op, {x}, span); \
876  } \
877  }

1116 template <typename ValueType>
1117 inline PrimExpr MakeConstScalar(DataType t, ValueType value, Span span = Span()) {
1118  if (t.is_int()) return IntImm(t, static_cast<int64_t>(value), span);
1119  if (t.is_uint()) {
1120  // Use IntImm if it is a small integer
1121  uint64_t uval = static_cast<uint64_t>(value);
1122  if (uval <= static_cast<uint64_t>(std::numeric_limits<int64_t>::max())) {
1123  return IntImm(t, static_cast<int64_t>(value), span);
1124  } else {
1125  uint64_t mask = (static_cast<uint64_t>(1) << 32U) - 1U;
1126  uint64_t low = uval & mask;
1127  uint64_t high = uval >> 32U;
1128  return LargeUIntImm(t, static_cast<int64_t>(low), static_cast<int64_t>(high), span);
1129  }
1130  }
1131  if (t.is_float() || t.is_bfloat16()) return FloatImm(t, static_cast<double>(value), span);
1132  // For now, we store const scalar values of custom datatypes within doubles; later, during the
1133  // datatypes lowering pass, we will lower the value to its true representation in the format
1134  // specified by the datatype.
1135  // TODO(gus) when do we need to start worrying about doubles not being precise enough?
1136  if (static_cast<uint8_t>(t.code()) >= static_cast<uint8_t>(DataType::kCustomBegin)) {
1137  return FloatImm(t, static_cast<double>(value), span);
1138  }
1139  LOG(FATAL) << "cannot make const for type " << t;
1140  return PrimExpr();
1141 }
