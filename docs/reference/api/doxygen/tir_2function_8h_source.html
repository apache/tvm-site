244 /*
245  * \brief Specialize parameters of PrimFunc.
246  * \param func The PrimFunc to be specialized.
247  * \param param_map The mapping from function params to the instance.
248  * \return The new function with parameter specialized.
249  * \note We can define a Meta TIR function with symbolic shape:
250  *
251  * \code
252  * @T.prim_func
253  * def mem_copy(a: T.handle, b: T.handle, m: T.int32, n: T.int32) -> None:
254  * A = T.match_buffer(a, (m, n), "float32")
255  * B = T.match_buffer(b, (m, n), "float32")
256  * for i, j in T.grid(m, n):
257  * with T.block():
258  * vi, vj = T.axis.remap("SS", [i, j])
259  * B[vi, vj] = A[vi, vj]
260  * \endcode
261  *
262  * Then we can make it specialized with given shapes or buffers.
263  *
264  * \code
265  * a, _, m, n = mem_copy.params
266  * func = mem_copy.specialize({a: tir.decl_buffer((16, 16))})
267  * # or
268  * func = mem_copy.specialize({n: 16, m: 16})
269  * \endcode
270  *
271  * \code {.language-id}
272  * @T.prim_func
273  * def mem_copy_16_16(a: T.handle, b: T.handle) -> None:
274  * A = T.match_buffer(a, (16, 16), "float32")
275  * B = T.match_buffer(b, (16, 16), "float32")
276  * for i, j in T.grid(16, 16):
277  * with T.block():
278  * vi, vj = T.axis.remap("SS", [i, j])
279  * B[vi, vj] = A[vi, vj]
280  * \endcode
281  */
