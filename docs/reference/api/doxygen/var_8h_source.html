339 /* \brief Allow tirx.Var as key in STL tables
340  *
341  * For most TIR expressions, it would be ambiguous whether the
342  * expression should follow reference equality or structural equality.
343  * This is not the case for variables, which do not contain nested
344  * internal structure, and are frequently used as keys in lookup
345  * tables.
346  *
347  * Providing `std::hash` and `std::equal_to` specializations for
348  * `tirx::Var` allows it to be used as a key in STL tables. For
349  * `PrimExpr`, the user must specify the type of equality used
350  * (e.g. `std::unordered_set<T, StructuralHash, StructuralEqual>` or
351  * `std::unordered_set<T, ffi::ObjectPtrHash, ffi::ObjectPtrEqual>`).
352  */
353 template <>
354 struct std::hash<tvm::tirx::Var> {
355  std::size_t operator()(const tvm::tirx::Var& var) const { return tvm::ffi::ObjectPtrHash()(var); }
356 };
357 
358 template <>
359 struct std::equal_to<tvm::tirx::Var> {
360  bool operator()(const tvm::tirx::Var& var_a, const tvm::tirx::Var& var_b) const {
361  return tvm::ffi::ObjectPtrEqual()(var_a, var_b);
362  }
363 };
