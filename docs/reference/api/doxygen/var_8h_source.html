342 /* \brief Allow tir.Var as key in STL tables
343  *
344  * For most TIR expressions, it would be ambiguous whether the
345  * expression should follow reference equality or structural equality.
346  * This is not the case for variables, which do not contain nested
347  * internal structure, and are frequently used as keys in lookup
348  * tables.
349  *
350  * Providing `std::hash` and `std::equal_to` specializations for
351  * `tir::Var` allows it to be used as a key in STL tables. For
352  * `PrimExpr`, the user must specify the type of equality used
353  * (e.g. `std::unordered_set<T, StructuralHash, StructuralEqual>` or
354  * `std::unordered_set<T, ObjectPtrHash, ObjectPtrEqual>`).
355  */
356 template <>
357 struct std::hash<tvm::tir::Var> {
358  std::size_t operator()(const tvm::tir::Var& var) const {
359  return tvm::runtime::ObjectPtrHash()(var);
360  }
361 };
362 
363 template <>
364 struct std::equal_to<tvm::tir::Var> {
365  bool operator()(const tvm::tir::Var& var_a, const tvm::tir::Var& var_b) const {
366  return tvm::runtime::ObjectPtrEqual()(var_a, var_b);
367  }
368 };
