265  }

272  return LiteralDoc(IntImm(DataType::Int(64), v), p);
273  }

280  return LiteralDoc(IntImm(DataType::Bool(), v), p);

302  std::string dtype = v.is_void() ? "void" : runtime::DLDataTypeToString(v);
303  return LiteralDoc::Str(dtype, p);
304  }
310  static LiteralDoc Device(const DLDevice& v, const Optional<ObjectPath>& p) {
