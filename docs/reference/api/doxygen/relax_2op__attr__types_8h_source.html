34 enum OpPatternKind {
35  // Elementwise operation
36  kElemWise = 0,
37  // Broadcasting operator, can always map output axis to the input in order.
38  // for example :code:`out[i, ax1, j, ax2] = input[i, j]`.
39  // Note that the axis need to be in order so transpose is not a bcast operator.
40  kBroadcast = 1,
41  // Injective operator, can always injectively map output axis to a single input axis.
42  // All injective operator can still be safely fused to injective and reduction.
43  kInjective = 2,
44  // Communicative reduction operator.
45  kCommReduce = 3,
46  // Complex operation, can still fuse elemwise operations into its output.
47  // but cannot chain another complex op
48  kOutEWiseFusable = 4,
49  // The pattern for tuple nodes. Can fuse into subsequent injective ops,
50  // but treated specially
51  kTuple = 7,
52  // Opaque operation, cannot fuse anything.
53  kOpaque = 8
54 };
