911 /*
912  * \brief Template function to convert Map to unordered_map
913  * Sometimes useful for API gluing when internal uses unordered_map
914  * \param dmap The container map
915  * \return The corresponding unordered_map.
916  * \tparam K the key of the Map.
917  * \tparam V the value of the Map.
918  */
919 template <typename K, typename V>
920 inline std::unordered_map<K, V> as_unordered_map(const ffi::Map<K, V>& dmap) {
