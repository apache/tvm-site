349 template <typename... Args>
350 DRef SessionObj::CallPacked(const DRef& func, Args&&... args) {
351  constexpr int offset = 3;
352  constexpr int kNumArgs = offset + sizeof...(Args);
353  ffi::AnyView packed_args[kNumArgs];
354  ffi::PackedArgs::Fill(packed_args,
355  /*.0=*/static_cast<int>(DiscoAction::kCallPacked), // action
356  /*.1=*/0, // reg_id, which will be updated by this->CallWithPacked
357  /*.2=*/func, // the function to be called
358  std::forward<Args>(args)...);
359  return this->CallWithPacked(ffi::PackedArgs(packed_args, kNumArgs));
360 }
