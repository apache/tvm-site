800  template <typename... Args>
801  static Stmt Flatten(Args&&... seq_args) {
802  Array<Stmt> seq;
803  runtime::detail::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);
804  if (seq.size() == 1) return seq[0];
805  return SeqStmt(seq);
806  }
808  class Flattener {
809  public:
810  explicit Flattener(Array<Stmt>* seq) : seq_(seq) {}
811 
812  void operator()(size_t i, const Stmt& stmt) const {
813  if (!stmt.defined()) return;
814  if (auto* op = stmt.as<SeqStmtNode>()) {
815  operator()(0, op->seq);
816  } else {
817  seq_->push_back(stmt);
818  }
819  }
820 
821  template <typename T>
822  void operator()(size_t i, const T& seq) const {
823  for (auto v : seq) {
824  this->operator()(0, v);
825  }
826  }
827 
828  private:
829  Array<Stmt>* seq_;
830  };
