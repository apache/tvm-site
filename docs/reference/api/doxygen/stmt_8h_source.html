534  size_t size() const { return operator->()->size(); }

559  template <typename... Args>
560  static Stmt Flatten(Args&&... seq_args) {
561  ffi::Array<Stmt> seq;
562 
563  ffi::details::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);
564 
565  if (seq.empty()) {
566  return Evaluate(0);
567  } else if (seq.size() == 1) {
568  return seq[0];
569  }
570 
571  // If the argument is a single SeqStmt argument with no
572  // flattening or unwrapping required, then we may
573  // return the SeqStmt as-is.
574  if constexpr (sizeof...(seq_args) == 1) {
575  if (auto opt = Flattener::AsSeqStmt(std::forward<Args>(seq_args)...)) {
576  SeqStmt original = opt.value();
577  bool all_same = [&]() {
578  if (original->seq.size() != seq.size()) {
579  return false;
580  }
581  for (size_t i = 0; i < seq.size(); i++) {
582  if (!original->seq[i].same_as(seq[i])) {
583  return false;
584  }
585  }
586  return true;
587  }();
588  if (all_same) {
589  return original;
590  }
591  }
592  }
593 
594  return SeqStmt(seq);
595  }

598  public:
599  explicit Flattener(ffi::Array<Stmt>* seq) : seq_(seq) {}

601  template <typename T>
602  static ffi::Optional<SeqStmt> AsSeqStmt(const T& t) {
603  if constexpr (std::is_same_v<T, SeqStmt>) {
604  return t;
605  }
606  if constexpr (!std::is_base_of_v<T, SeqStmt>) {
607  return std::nullopt;
608  }
609  if constexpr (std::is_base_of_v<Stmt, T>) {
610  if (const SeqStmtNode* ptr = t.template as<SeqStmtNode>()) {
611  return ffi::GetRef<SeqStmt>(ptr);
612  } else {
613  return std::nullopt;
614  }
615  }
616  return std::nullopt;
617  }
618 
619  template <typename T>
620  void operator()(size_t i, const T& stmt_or_seq) const {
621  if constexpr (std::is_base_of_v<ObjectRef, T>) {
622  // Early bail-out, applicable to any ObjectRef
623  if (!stmt_or_seq.defined()) {
624  return;
625  }
626  }
627 
628  if constexpr (std::is_same_v<T, SeqStmt>) {
629  // Static type-checking for a SeqStmt that could be flattened.
630  (*this)(0, stmt_or_seq->seq);
631  return;
632  }
633 
634  if constexpr (std::is_base_of_v<T, SeqStmt>) {
635  // Dynamic type-checking for a SeqStmt that could be
636  // flattened.
637  if (auto* op = stmt_or_seq.template as<SeqStmtNode>()) {
638  operator()(0, op->seq);
639  return;
640  }
641  }
642 
643  if constexpr (std::is_base_of_v<T, Evaluate>) {
644  // Evaluate(0) is used to represent a no-op, and may be
645  // generated by previous calls to SeqStmt::Flatten(). These
646  // should be removed to ensure that Flatten(a+b) is equivalent
647  // to Flatten(Flatten(a), Flatten(b)).
648  if (auto* op = stmt_or_seq.template as<EvaluateNode>()) {
649  if (auto* as_int = op->value.template as<IntImmNode>(); as_int && as_int->value == 0) {
650  return;
651  }
652  }
653  }
654 
655  if constexpr (std::is_base_of_v<Stmt, T>) {
656  // Any other Stmt type just gets appended.
657  seq_->push_back(stmt_or_seq);
658  } else {
659  // Anything else is treated as an iterable of Stmt.
660  for (auto v : stmt_or_seq) {
661  this->operator()(0, v);
662  }
663  }
664  }
665 
666  private:
667  ffi::Array<Stmt>* seq_;
668  };
