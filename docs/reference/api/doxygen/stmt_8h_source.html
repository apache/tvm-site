745  template <typename... Args>

747  Array<Stmt> seq;
748  runtime::detail::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);

758  if (!stmt.defined()) return;
759  if (auto* op = stmt.as<SeqStmtNode>()) {
760  operator()(0, op->seq);

762  seq_->push_back(stmt);
763  }
764  }
