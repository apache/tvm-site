411  template <typename... Args>
412  static Stmt Flatten(Args&&... seq_args) {
413  ffi::Array<Stmt> seq;
414 
415  ffi::details::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);
416 
417  if (seq.empty()) {
418  return Evaluate(0);
419  } else if (seq.size() == 1) {
420  return seq[0];
421  }
422 
423  // If the argument is a single SeqStmt argument with no
424  // flattening or unwrapping required, then we may
425  // return the SeqStmt as-is.
426  if constexpr (sizeof...(seq_args) == 1) {
427  if (auto opt = Flattener::AsSeqStmt(std::forward<Args>(seq_args)...)) {
428  SeqStmt original = opt.value();
429  bool all_same = [&]() {
430  if (original->seq.size() != seq.size()) {
431  return false;
432  }
433  for (size_t i = 0; i < seq.size(); i++) {
434  if (!original->seq[i].same_as(seq[i])) {
435  return false;
436  }
437  }
438  return true;
439  }();
440  if (all_same) {
441  return original;
442  }
443  }
444  }
445 
446  return SeqStmt(seq);
447  }

453  template <typename T>
454  static ffi::Optional<SeqStmt> AsSeqStmt(const T& t) {
455  if constexpr (std::is_same_v<T, SeqStmt>) {
456  return t;
457  }
458  if constexpr (!std::is_base_of_v<T, SeqStmt>) {
459  return std::nullopt;
460  }
461  if constexpr (std::is_base_of_v<Stmt, T>) {
462  if (const SeqStmtNode* ptr = t.template as<SeqStmtNode>()) {
463  return ffi::GetRef<SeqStmt>(ptr);
464  } else {
465  return std::nullopt;
466  }
467  }
468  return std::nullopt;
469  }

471  template <typename T>
472  void operator()(size_t i, const T& stmt_or_seq) const {
473  if constexpr (std::is_base_of_v<ObjectRef, T>) {
474  // Early bail-out, applicable to any ObjectRef
475  if (!stmt_or_seq.defined()) {
476  return;
477  }
478  }
479 
480  if constexpr (std::is_same_v<T, SeqStmt>) {
481  // Static type-checking for a SeqStmt that could be flattened.
482  (*this)(0, stmt_or_seq->seq);
483  return;
484  }
485 
486  if constexpr (std::is_base_of_v<T, SeqStmt>) {
487  // Dynamic type-checking for a SeqStmt that could be
488  // flattened.
489  if (auto* op = stmt_or_seq.template as<SeqStmtNode>()) {
490  operator()(0, op->seq);
491  return;
492  }
493  }
494 
495  if constexpr (std::is_base_of_v<T, Evaluate>) {
496  // Evaluate(0) is used to represent a no-op, and may be
497  // generated by previous calls to SeqStmt::Flatten(). These
498  // should be removed to ensure that Flatten(a+b) is equivalent
499  // to Flatten(Flatten(a), Flatten(b)).
500  if (auto* op = stmt_or_seq.template as<EvaluateNode>()) {
501  if (auto* as_int = op->value.template as<IntImmNode>(); as_int && as_int->value == 0) {
502  return;
503  }
504  }
505  }
506 
507  if constexpr (std::is_base_of_v<Stmt, T>) {
508  // Any other Stmt type just gets appended.
509  seq_->push_back(stmt_or_seq);
510  } else {
511  // Anything else is treated as an iterable of Stmt.
512  for (auto v : stmt_or_seq) {
513  this->operator()(0, v);
514  }
515  }
516  }
517 
518  private:
519  ffi::Array<Stmt>* seq_;
520  };
