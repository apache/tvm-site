51 // ---- Arithmetic types ----
52 template <typename T>
53 struct Serializer<T, std::enable_if_t<std::is_arithmetic_v<T>>> {
54  static constexpr bool enabled = true;
55 
56  static void Write(Stream* strm, const T& data) {
57  if constexpr (TVM_FFI_IO_NO_ENDIAN_SWAP) {
58  strm->Write(&data, sizeof(T));
59  } else {
60  T copy = data;
61  ffi::ByteSwap(&copy, sizeof(T), 1);
62  strm->Write(&copy, sizeof(T));
63  }
64  }
65 
66  static bool Read(Stream* strm, T* data) {
67  bool ok = strm->Read(data, sizeof(T)) == sizeof(T);
68  if constexpr (!TVM_FFI_IO_NO_ENDIAN_SWAP) {
69  ffi::ByteSwap(data, sizeof(T), 1);
70  }
71  return ok;
72  }
73 };

119 template <typename T>
120 struct Serializer<std::vector<T>> {
121  static constexpr bool enabled = true;
122 
123  static void Write(Stream* strm, const std::vector<T>& vec) {
124  uint64_t sz = static_cast<uint64_t>(vec.size());
125  Serializer<uint64_t>::Write(strm, sz);
126  if constexpr (std::is_arithmetic_v<T> && TVM_FFI_IO_NO_ENDIAN_SWAP) {
127  if (sz != 0) {
128  strm->Write(vec.data(), sizeof(T) * vec.size());
129  }
130  } else {
131  for (const auto& v : vec) {
132  Serializer<T>::Write(strm, v);
133  }
134  }
135  }
136 
137  static bool Read(Stream* strm, std::vector<T>* vec) {
138  uint64_t sz;
139  if (!Serializer<uint64_t>::Read(strm, &sz)) return false;
140  vec->resize(static_cast<size_t>(sz));
141  if constexpr (std::is_arithmetic_v<T> && TVM_FFI_IO_NO_ENDIAN_SWAP) {
142  if (sz != 0) {
143  size_t nbytes = sizeof(T) * static_cast<size_t>(sz);
144  return strm->Read(vec->data(), nbytes) == nbytes;
145  }
146  return true;
147  } else {
148  for (size_t i = 0; i < static_cast<size_t>(sz); ++i) {
149  if (!Serializer<T>::Read(strm, &(*vec)[i])) return false;
150  }
151  return true;
152  }
153  }
