249  template <typename ValueType>
250  static int32_t RegisterConfigOption(const char* key) {
251  // NOTE: we could further update the function later.
252  if constexpr (std::is_base_of_v<ffi::ObjectRef, ValueType>) {
253  int32_t tindex = ffi::TypeToRuntimeTypeIndex<ValueType>::v();
254  auto type_key = ffi::TypeIndexToTypeKey(tindex);
255  auto legalization = [=](ffi::Any value) -> ffi::Any {
256  if (auto opt_map = value.try_cast<ffi::Map<ffi::String, ffi::Any>>()) {
257  return ffi::reflection::ObjectCreator(type_key)(opt_map.value());
258  } else {
259  auto opt_val = value.try_cast<ValueType>();
260  if (!opt_val.has_value()) {
261  TVM_FFI_THROW(AttributeError)
262  << "Expect config " << key << " to have type " << type_key << ", but instead get "
263  << ffi::details::AnyUnsafe::GetMismatchTypeInfo<ValueType>(value);
264  }
265  return *opt_val;
266  }
267  };
268  RegisterConfigOption(key, type_key, legalization);
269  } else {
270  // non-object type, do not support implicit conversion from map
271  std::string type_str = ffi::TypeTraits<ValueType>::TypeStr();
272  auto legalization = [=](ffi::Any value) -> ffi::Any {
273  auto opt_val = value.try_cast<ValueType>();
274  if (!opt_val.has_value()) {
275  TVM_FFI_THROW(AttributeError)
276  << "Expect config " << key << " to have type " << type_str << ", but instead get "
277  << ffi::details::AnyUnsafe::GetMismatchTypeInfo<ValueType>(value);
278  } else {
279  return *opt_val;
280  }
281  };
282  RegisterConfigOption(key, type_str, legalization);
283  }
284  return 0;
285  }
