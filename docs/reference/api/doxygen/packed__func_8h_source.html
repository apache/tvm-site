48 inline TVMFFIAny LegacyTVMArgValueToFFIAny(TVMValue value, int type_code) {
49  TVMFFIAny res;
50  // clear first to ensure consistent hash
51  res.v_uint64 = 0;
52  switch (type_code) {
53  case kTVMArgInt: {
54  res.type_index = ffi::TypeIndex::kTVMFFIInt;
55  res.v_int64 = value.v_int64;
56  return res;
57  }
58  case kTVMArgFloat: {
59  res.type_index = ffi::TypeIndex::kTVMFFIFloat;
60  res.v_float64 = value.v_float64;
61  return res;
62  }
63  case kTVMOpaqueHandle: {
64  res.type_index = ffi::TypeIndex::kTVMFFIOpaquePtr;
65  res.v_ptr = value.v_handle;
66  return res;
67  }
68  case kTVMNullptr: {
69  res.type_index = ffi::TypeIndex::kTVMFFINone;
70  return res;
71  }
72  case kTVMDataType: {
73  res.type_index = ffi::TypeIndex::kTVMFFIDataType;
74  res.v_dtype = value.v_type;
75  return res;
76  }
77  case kDLDevice: {
78  res.type_index = ffi::TypeIndex::kTVMFFIDevice;
79  res.v_device = value.v_device;
80  return res;
81  }
82  case kTVMDLTensorHandle: {
83  res.type_index = ffi::TypeIndex::kTVMFFIDLTensorPtr;
84  res.v_ptr = value.v_handle;
85  return res;
86  }
87  case kTVMObjectHandle: {
88  res.v_obj = static_cast<TVMFFIObject*>(value.v_handle);
89  res.type_index = res.v_obj->type_index;
90  return res;
91  }
92  case kTVMModuleHandle: {
93  res.type_index = ffi::TypeIndex::kTVMFFIModule;
94  res.v_obj = static_cast<TVMFFIObject*>(value.v_handle);
95  return res;
96  }
97  case kTVMPackedFuncHandle: {
98  res.type_index = ffi::TypeIndex::kTVMFFIFunction;
99  res.v_obj = static_cast<TVMFFIObject*>(value.v_handle);
100  return res;
101  }
102  case kTVMStr: {
103  res.type_index = ffi::TypeIndex::kTVMFFIRawStr;
104  res.v_c_str = value.v_str;
105  return res;
106  }
107  case kTVMBytes: {
108  res.type_index = ffi::TypeIndex::kTVMFFIByteArrayPtr;
109  res.v_ptr = value.v_handle;
110  return res;
111  }
112  case kTVMNDArrayHandle: {
113  res.type_index = ffi::TypeIndex::kTVMFFINDArray;
114  res.v_obj = reinterpret_cast<TVMFFIObject*>(TVMArrayHandleToObjectHandle(value.v_handle));
115  return res;
116  }
117  case kTVMArgBool: {
118  res.type_index = ffi::TypeIndex::kTVMFFIBool;
119  res.v_int64 = value.v_int64;
120  return res;
121  }
122  case kTVMObjectRValueRefArg: {
123  res.type_index = ffi::TypeIndex::kTVMFFIObjectRValueRef;
124  res.v_ptr = value.v_handle;
125  return res;
126  }
127  default: {
128  LOG(FATAL) << "Unsupported type code: " << type_code;
129  TVM_FFI_UNREACHABLE();
130  }
131  }
132 }

150 /*
151  * \brief Convert AnyView to legacy TVMValue and type_code
152  * \param src The AnyView to convert
153  * \param value The TVMValue to store the result
154  * \param type_code The type code to store the result
155  * \note This routine is not fastest, but serves purpose to do transition of ABI.
156  */
157 inline void AnyViewToLegacyTVMArgValue(TVMFFIAny src, TVMValue* value, int* type_code) {
158  switch (src.type_index) {
159  case ffi::TypeIndex::kTVMFFIBool: {
160  type_code[0] = kTVMArgBool;
161  value[0].v_int64 = src.v_int64;
162  break;
163  }
164  case ffi::TypeIndex::kTVMFFIInt: {
165  type_code[0] = kDLInt;
166  value[0].v_int64 = src.v_int64;
167  break;
168  }
169  case ffi::TypeIndex::kTVMFFIFloat: {
170  type_code[0] = kDLFloat;
171  value[0].v_float64 = src.v_float64;
172  break;
173  }
174  case ffi::TypeIndex::kTVMFFIOpaquePtr: {
175  type_code[0] = kTVMOpaqueHandle;
176  value[0].v_handle = src.v_ptr;
177  break;
178  }
179  case ffi::TypeIndex::kTVMFFINone: {
180  type_code[0] = kTVMNullptr;
181  break;
182  }
183  case ffi::TypeIndex::kTVMFFIDataType: {
184  type_code[0] = kTVMDataType;
185  value[0].v_type = src.v_dtype;
186  break;
187  }
188  case ffi::TypeIndex::kTVMFFIDevice: {
189  type_code[0] = kDLDevice;
190  value[0].v_device = src.v_device;
191  break;
192  }
193  case ffi::TypeIndex::kTVMFFIDLTensorPtr: {
194  type_code[0] = kTVMDLTensorHandle;
195  value[0].v_handle = src.v_ptr;
196  break;
197  }
198  case ffi::TypeIndex::kTVMFFIRawStr: {
199  type_code[0] = kTVMStr;
200  value[0].v_str = src.v_c_str;
201  break;
202  }
203  case ffi::TypeIndex::kTVMFFIByteArrayPtr: {
204  type_code[0] = kTVMBytes;
205  value[0].v_handle = src.v_ptr;
206  break;
207  }
208  case ffi::TypeIndex::kTVMFFINDArray: {
209  type_code[0] = kTVMNDArrayHandle;
210  value[0].v_handle = ObjectHandleToTVMArrayHandle(reinterpret_cast<Object*>(src.v_obj));
211  break;
212  }
213  case ffi::TypeIndex::kTVMFFIModule: {
214  type_code[0] = kTVMModuleHandle;
215  value[0].v_handle = src.v_obj;
216  break;
217  }
218  case ffi::TypeIndex::kTVMFFIFunction: {
219  type_code[0] = kTVMPackedFuncHandle;
220  value[0].v_handle = src.v_obj;
221  break;
222  }
223  case ffi::TypeIndex::kTVMFFIObjectRValueRef: {
224  type_code[0] = kTVMObjectRValueRefArg;
225  value[0].v_handle = src.v_ptr;
226  break;
227  }
228  default: {
229  if (src.type_index >= ffi::TypeIndex::kTVMFFIStaticObjectBegin) {
230  type_code[0] = kTVMObjectHandle;
231  value[0].v_handle = src.v_obj;
232  break;
233  }
234  LOG(FATAL) << "Unsupported type index: " << src.type_index;
235  }
236  }
237 }
