399  template <typename... Args>
400  static Stmt Flatten(Args&&... seq_args) {
401  ffi::Array<Stmt> seq;
402 
403  ffi::details::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);
404 
405  if (seq.empty()) {
406  return Evaluate(0);
407  } else if (seq.size() == 1) {
408  return seq[0];
409  }
410 
411  // If the argument is a single SeqStmt argument with no
412  // flattening or unwrapping required, then we may
413  // return the SeqStmt as-is.
414  if constexpr (sizeof...(seq_args) == 1) {
415  if (auto opt = Flattener::AsSeqStmt(std::forward<Args>(seq_args)...)) {
416  SeqStmt original = opt.value();
417  bool all_same = [&]() {
418  if (original->seq.size() != seq.size()) {
419  return false;
420  }
421  for (size_t i = 0; i < seq.size(); i++) {
422  if (!original->seq[i].same_as(seq[i])) {
423  return false;
424  }
425  }
426  return true;
427  }();
428  if (all_same) {
429  return original;
430  }
431  }
432  }
433 
434  return SeqStmt(seq);
435  }

438  public:
439  explicit Flattener(ffi::Array<Stmt>* seq) : seq_(seq) {}

441  template <typename T>
442  static ffi::Optional<SeqStmt> AsSeqStmt(const T& t) {
443  if constexpr (std::is_same_v<T, SeqStmt>) {
444  return t;
445  }
446  if constexpr (!std::is_base_of_v<T, SeqStmt>) {
447  return std::nullopt;
448  }
449  if constexpr (std::is_base_of_v<Stmt, T>) {
450  if (const SeqStmtNode* ptr = t.template as<SeqStmtNode>()) {
451  return ffi::GetRef<SeqStmt>(ptr);
452  } else {
453  return std::nullopt;
454  }
455  }
456  return std::nullopt;
457  }
458 
459  template <typename T>
460  void operator()(size_t i, const T& stmt_or_seq) const {
461  if constexpr (std::is_base_of_v<ffi::ObjectRef, T>) {
462  // Early bail-out, applicable to any ffi::ObjectRef
463  if (!stmt_or_seq.defined()) {
464  return;
465  }
466  }
467 
468  if constexpr (std::is_same_v<T, SeqStmt>) {
469  // Static type-checking for a SeqStmt that could be flattened.
470  (*this)(0, stmt_or_seq->seq);
471  return;
472  }
473 
474  if constexpr (std::is_base_of_v<T, SeqStmt>) {
475  // Dynamic type-checking for a SeqStmt that could be
476  // flattened.
477  if (auto* op = stmt_or_seq.template as<SeqStmtNode>()) {
478  operator()(0, op->seq);
479  return;
480  }
481  }
482 
483  if constexpr (std::is_base_of_v<T, Evaluate>) {
484  // Evaluate(0) is used to represent a no-op, and may be
485  // generated by previous calls to SeqStmt::Flatten(). These
486  // should be removed to ensure that Flatten(a+b) is equivalent
487  // to Flatten(Flatten(a), Flatten(b)).
488  if (auto* op = stmt_or_seq.template as<EvaluateNode>()) {
489  if (auto* as_int = op->value.template as<IntImmNode>(); as_int && as_int->value == 0) {
490  return;
491  }
492  }
493  }
494 
495  if constexpr (std::is_base_of_v<Stmt, T>) {
496  // Any other Stmt type just gets appended.
497  seq_->push_back(stmt_or_seq);
498  } else {
499  // Anything else is treated as an iterable of Stmt.
500  for (auto v : stmt_or_seq) {
501  this->operator()(0, v);
502  }
503  }
504  }
505 
506  private:
507  ffi::Array<Stmt>* seq_;
508  };

510  TVM_FFI_DEFINE_OBJECT_REF_METHODS_NULLABLE(SeqStmt, Stmt, SeqStmtNode);
511  TVM_DEFINE_OBJECT_REF_COW_METHOD(SeqStmtNode);
512 };
