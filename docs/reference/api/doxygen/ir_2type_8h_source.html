416 class FuncTypeNode : public TypeNode {
417  public:
419  Array<Type> arg_types;

423  // For normal functions, the following two fields will be empty.

430  Array<TypeConstraint> type_constraints;
431 
432  void VisitAttrs(AttrVisitor* v) {
433  v->Visit("arg_types", &arg_types);
434  v->Visit("ret_type", &ret_type);
435  v->Visit("type_params", &type_params);
436  v->Visit("type_constraints", &type_constraints);
437  v->Visit("span", &span);
438  }
439 
440  bool SEqualReduce(const FuncTypeNode* other, SEqualReducer equal) const {
441  // type params first as they defines type vars.
442  return equal.DefEqual(type_params, other->type_params) && equal(arg_types, other->arg_types) &&
443  equal(ret_type, other->ret_type) && equal(type_constraints, other->type_constraints);
444  }
445 
446  void SHashReduce(SHashReducer hash_reduce) const {
447  hash_reduce.DefHash(type_params);
448  hash_reduce(arg_types);
449  hash_reduce(ret_type);
450  hash_reduce(type_constraints);
451  }
