321  enum Extension {
322  // No extensions enabled
323  kNone = 0,
324 
325  /* When simplifying an inequality, attempt to use scope-based knowns.
326  *
327  * Example:
328  * if_then_else(i<j && j<k, i<k, false) => if_then_else(i<j && j<k, true, false)
329  */
330  kTransitivelyProveInequalities = (1 << 0),
331 
332  /* When simplifying a boolean expression, convert to an AND of ORs
333  * (conjunctive normal form).
334  *
335  * Example:
336  * (a && b) || c => (a || c) && (b || c)
337  */
338  kConvertBooleanToAndOfOrs = (1 << 1),
339 
340  /* When simplifying a boolean AND or a boolean OR, simplify each
341  * branch under the assumption that the other branch does not
342  * already dominate the result. That is, simplify each branch of
343  * (A && B) under the assumption that the other branch is true,
344  * and simplify each branch of (A || B) under the assumption that
345  * the other branch is false.
346  *
347  * Example:
348  * (n < 10) && (n < 5) => (n < 10)
349  * (n < 10) || (n < 5) => (n < 5)
350  */
351  kApplyConstraintsToBooleanBranches = (1 << 2),
352 
353  /* Special handling for expressions `(A+B)*C < (A*B)*D`
354  *
355  * Expressions of the form `(A+B)*C < (A*B)*D` can occur occur
356  * when comparing the number of operations required for two
357  * different orderings in which matrix multiplications can be
358  * performed. Proving or disproving this conditional allows an
359  * optimal order of execution to be selected, even for dynamic
360  * argument shapes.
361  *
362  * The default behavior of `ConstIntBounds` assumes that each term
363  * in an expression is independent, and is insufficient to prove
364  * these inequalities. For example, the maximum value of `(A+B)*C
365  * - (A*B)*D` is determined by taking the maximum value of
366  * `(A+B)*C` and subtracting the minimum value of `(A*B)*D`.
367  * While this algorithm can be applied in all cases, the bound it
368  * provides is looser than strictly required.
369  *
370  * This extension adds a check for this case. When `A`, `B`, `C`,
371  * and `D` are all positive values, as is the case for tensor
372  * shapes, the inequality can be written as `1/A + 1/B < D/C`. If
373  * this inequality holds for the minimum values of `A`, `B`, and
374  * `D`, along with the maximum value of `C`, then the inequality
375  * holds for all values.
376  *
377  * This extension requires little to no performance overhead, and
378  * may be enabled by default in future releases.
379  */
380  kComparisonOfProductAndSum = (1 << 3),
381  };
