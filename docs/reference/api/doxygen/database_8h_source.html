235 
236  void VisitAttrs(tvm::AttrVisitor* v) {
237  // PackedFuncs are all not visited, because the reflection system doesn't take care of them,
238  // so it cannot be accessible on the python side. If there is such need from the future,
239  // we can then add corresponding accessor methods to help access on python.
240  // `f_has_workload` is not visited
241  // `f_commit_workload` is not visited
242  // `f_commit_tuning_record` is not visited
243  // `f_get_top_k` is not visited
244  // `f_size` is not visited
245  }
246 
247  bool HasWorkload(const IRModule& mod) final {
248  ICHECK(f_has_workload != nullptr) << "PyDatabase's HasWorkload method not implemented!";
249  return f_has_workload(mod);
250  }
251 
252  Workload CommitWorkload(const IRModule& mod) final {
253  ICHECK(f_commit_workload != nullptr) << "PyDatabase's CommitWorkload method not implemented!";
254  return f_commit_workload(mod);
255  }
256 
257  void CommitTuningRecord(const TuningRecord& record) final {
258  ICHECK(f_commit_tuning_record != nullptr)
259  << "PyDatabase's CommitTuningRecord method not implemented!";
260  f_commit_tuning_record(record);
261  }
262 
263  Array<TuningRecord> GetTopK(const Workload& workload, int top_k) final {
264  ICHECK(f_get_top_k != nullptr) << "PyDatabase's GetTopK method not implemented!";
265  return f_get_top_k(workload, top_k);
266  }
267 
268  int64_t Size() final {
269  ICHECK(f_size != nullptr) << "PyDatabase's Size method not implemented!";
270  return f_size();
