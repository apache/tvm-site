386  static void RegisterReflection() {
387  // ffi::Functions are all not registered, because the reflection system doesn't take care of
388  // them, so it cannot be accessible on the python side. If there is such need from the future,
389  // we can then add corresponding accessor methods to help access on python.
390  // `f_has_workload` is not registered
391  // `f_commit_workload` is not registered
392  // `f_commit_tuning_record` is not registered
393  // `f_get_top_k` is not registered
394  // `f_get_all_tuning_records` is not registered
395  // `f_query_tuning_record` is not registered
396  // `f_query_schedule` is not registered
397  // `f_query_ir_module` is not registered
398  // `f_size` is not registered
399  }
400
