277  FHasWorkload f_has_workload;

281  FCommitTuningRecord f_commit_tuning_record;
283  FGetTopK f_get_top_k;
285  FGetAllTuningRecords f_get_all_tuning_records;
287  FSize f_size;
288 
289  void VisitAttrs(tvm::AttrVisitor* v) {
290  // PackedFuncs are all not visited, because the reflection system doesn't take care of them,
291  // so it cannot be accessible on the python side. If there is such need from the future,
292  // we can then add corresponding accessor methods to help access on python.
293  // `f_has_workload` is not visited
294  // `f_commit_workload` is not visited
295  // `f_commit_tuning_record` is not visited
296  // `f_get_top_k` is not visited
297  // `f_get_all_tuning_records` is not visited
298  // `f_size` is not visited
299  }
300 
301  bool HasWorkload(const IRModule& mod) final {
302  ICHECK(f_has_workload != nullptr) << "PyDatabase's HasWorkload method not implemented!";
303  return f_has_workload(mod);
304  }
305 
306  Workload CommitWorkload(const IRModule& mod) final {
307  ICHECK(f_commit_workload != nullptr) << "PyDatabase's CommitWorkload method not implemented!";
308  return f_commit_workload(mod);
309  }
310 
311  void CommitTuningRecord(const TuningRecord& record) final {
312  ICHECK(f_commit_tuning_record != nullptr)
313  << "PyDatabase's CommitTuningRecord method not implemented!";
314  f_commit_tuning_record(record);
315  }
316 
317  Array<TuningRecord> GetTopK(const Workload& workload, int top_k) final {
318  ICHECK(f_get_top_k != nullptr) << "PyDatabase's GetTopK method not implemented!";
319  return f_get_top_k(workload, top_k);
320  }
321 
322  Array<TuningRecord> GetAllTuningRecords() final {
323  ICHECK(f_get_all_tuning_records != nullptr)
324  << "PyDatabase's GetAllTuningRecords method not implemented!";
325  return f_get_all_tuning_records();
326  }
327 
328  int64_t Size() final {
329  ICHECK(f_size != nullptr) << "PyDatabase's Size method not implemented!";
330  return f_size();
331  }

333  static constexpr const char* _type_key = "meta_schedule.PyDatabase";
