297 template <typename T, typename TraitName>
298 inline ReflectionVTable::Registry ReflectionVTable::Register() {
299  uint32_t tindex = T::RuntimeTypeIndex();
300  if (tindex >= fcreate_.size()) {
301  fcreate_.resize(tindex + 1, nullptr);
302  frepr_bytes_.resize(tindex + 1, nullptr);
303  fsequal_reduce_.resize(tindex + 1, nullptr);
304  fshash_reduce_.resize(tindex + 1, nullptr);
305  }
306  // functor that implements the redirection.
307  fsequal_reduce_[tindex] = ::tvm::detail::SelectSEqualReduce<T, TraitName>::SEqualReduce;
308 
309  fshash_reduce_[tindex] = ::tvm::detail::SelectSHashReduce<T, TraitName>::SHashReduce;
310 
311  return Registry(this, tindex);
312 }
