370 template <typename T, typename TraitName>
371 inline ReflectionVTable::Registry ReflectionVTable::Register() {
372  uint32_t tindex = T::RuntimeTypeIndex();
373  if (tindex >= fvisit_attrs_.size()) {
374  fvisit_attrs_.resize(tindex + 1, nullptr);
375  fcreate_.resize(tindex + 1, nullptr);
376  frepr_bytes_.resize(tindex + 1, nullptr);
377  fsequal_reduce_.resize(tindex + 1, nullptr);
378  fshash_reduce_.resize(tindex + 1, nullptr);
379  }
380  // functor that implements the redirection.
381  fvisit_attrs_[tindex] = ::tvm::detail::SelectVisitAttrs<T, TraitName>::VisitAttrs;
382 
383  fsequal_reduce_[tindex] = ::tvm::detail::SelectSEqualReduce<T, TraitName>::SEqualReduce;
384 
385  fshash_reduce_[tindex] = ::tvm::detail::SelectSHashReduce<T, TraitName>::SHashReduce;
386 
387  return Registry(this, tindex);
388 }

390 inline void ReflectionVTable::VisitAttrs(Object* self, AttrVisitor* visitor) const {
391  uint32_t tindex = self->type_index();
392  if (tindex >= fvisit_attrs_.size() || fvisit_attrs_[tindex] == nullptr) {
393  return;

401  if (repr_bytes != nullptr) {
