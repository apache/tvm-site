64  static TRandState DeviceRandom() {
65  std::random_device rd;

79  result_type operator()() {
80  (*rand_state_ptr_) = ((*rand_state_ptr_) * multiplier + increment) % modulus;
81  return *rand_state_ptr_;
82  }
88  static TRandState NormalizeSeed(TRandState rand_state) {
89  if (rand_state == -1) {
90  rand_state = DeviceRandom();
91  } else {
92  rand_state %= modulus;
93  }
94  if (rand_state == 0) {
95  rand_state = 1;
96  }
97  if (rand_state < 0) {
98  TVM_FFI_THROW(ValueError) << "Random seed must be non-negative";
99  }
100  return rand_state;
101  }

107  TVM_FFI_ICHECK(rand_state_ptr_ != nullptr);
108  *rand_state_ptr_ = NormalizeSeed(rand_state);
109  }
