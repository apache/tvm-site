133  bool SEqualReduce(const BufferNode* other, SEqualReducer equal) const {
134  // Use DefEqual as buffer can define variables in its semantics,
135  // skip name as name is not important.
136  return equal.DefEqual(data, other->data) && equal(dtype, other->dtype) &&
137  equal.DefEqual(shape, other->shape) && equal.DefEqual(strides, other->strides) &&
138  equal.DefEqual(axis_separators, other->axis_separators) &&
