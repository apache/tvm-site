450 template <typename FCheckVisited, typename FVisitLeaf, typename FExpandExpr>
451 void ExpandDataflow(Expr expr, FCheckVisited fcheck_visited, FVisitLeaf fvisit_leaf,
452  FExpandExpr fexpand_expr) {
453  std::deque<v_info> stack;
454  auto fpush_to_stack = [&fcheck_visited, &stack](const Expr& expr) {
455  if (!fcheck_visited(expr)) {
456  stack.emplace_front(v_info(expr));
457  }
458  };
459 
460  fpush_to_stack(expr);
461  while (stack.size() > 0) {
462  v_info* front = &stack.front();
463  if (fcheck_visited(front->node)) {
464  stack.pop_front();
465  } else if (front->children_expanded) {
466  fvisit_leaf(front->node);
467  // TODO(d-smirnov): this is for compatibility with current implementation of MixedModeVisitor
468  stack.pop_front();
469  } else {
470  front->children_expanded = true;
471  for (auto e : fexpand_expr(front->node)) {
472  fpush_to_stack(e);
473  }
474  }
475  }
476 }

478 template <typename FCheckVisited, typename FVisitLeaf>
479 void ExpandDataflow(Expr expr, FCheckVisited fcheck_visited, FVisitLeaf fvisit_leaf) {
480  auto fexpand_expr = [](const Expr& expr) {
481  std::vector<Expr> result;
482  if (const CallNode* op = expr.as<CallNode>()) {
483  if (op->op == Op::Get("call_lowered")) {
484  // Ignore the intermediate tuple since this is purely a calling-convention detail
485  const auto* tuple_args = op->args[1].as<TupleNode>();
486  ICHECK(tuple_args)
487  << "Expected second arg to call_lowered to be a Tuple of input arguments.";
488  for (auto it = tuple_args->fields.rbegin(); it != tuple_args->fields.rend(); ++it) {
489  result.push_back(*it);
490  }
491  result.push_back(op->args[0]);
492  } else {
493  for (auto it = op->args.rbegin(); it != op->args.rend(); ++it) {
494  result.push_back(*it);
495  }
496  }
497  result.push_back(op->op);
498  } else if (const TupleNode* op = expr.as<TupleNode>()) {
499  for (auto it = op->fields.rbegin(); it != op->fields.rend(); ++it) {
500  result.push_back(*it);
501  }
502  } else if (const TupleGetItemNode* op = expr.as<TupleGetItemNode>()) {
503  result.push_back(op->tuple);
504  }
505  return result;
506  };
507  ExpandDataflow(expr, fcheck_visited, fvisit_leaf, fexpand_expr);
508 }
