404  template <typename... Args>
405  static Stmt Flatten(Args&&... seq_args) {
406  ffi::Array<Stmt> seq;
407 
408  ffi::details::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);
409 
410  if (seq.empty()) {
411  return Evaluate(0);
412  } else if (seq.size() == 1) {
413  return seq[0];
414  }
415 
416  // If the argument is a single SeqStmt argument with no
417  // flattening or unwrapping required, then we may
418  // return the SeqStmt as-is.
419  if constexpr (sizeof...(seq_args) == 1) {
420  if (auto opt = Flattener::AsSeqStmt(std::forward<Args>(seq_args)...)) {
421  SeqStmt original = opt.value();
422  bool all_same = [&]() {
423  if (original->seq.size() != seq.size()) {
424  return false;
425  }
426  for (size_t i = 0; i < seq.size(); i++) {
427  if (!original->seq[i].same_as(seq[i])) {
428  return false;
429  }
430  }
431  return true;
432  }();
433  if (all_same) {
434  return original;
435  }
436  }
437  }
438 
439  return SeqStmt(seq);
440  }

446  template <typename T>
447  static ffi::Optional<SeqStmt> AsSeqStmt(const T& t) {
448  if constexpr (std::is_same_v<T, SeqStmt>) {
449  return t;
450  }
451  if constexpr (!std::is_base_of_v<T, SeqStmt>) {
452  return std::nullopt;
453  }
454  if constexpr (std::is_base_of_v<Stmt, T>) {
455  if (const SeqStmtNode* ptr = t.template as<SeqStmtNode>()) {
456  return ffi::GetRef<SeqStmt>(ptr);
457  } else {
458  return std::nullopt;
459  }
460  }
461  return std::nullopt;
462  }

464  template <typename T>
465  void operator()(size_t i, const T& stmt_or_seq) const {
466  if constexpr (std::is_base_of_v<ObjectRef, T>) {
467  // Early bail-out, applicable to any ObjectRef
468  if (!stmt_or_seq.defined()) {
469  return;
470  }
471  }
472 
473  if constexpr (std::is_same_v<T, SeqStmt>) {
474  // Static type-checking for a SeqStmt that could be flattened.
475  (*this)(0, stmt_or_seq->seq);
476  return;
477  }
478 
479  if constexpr (std::is_base_of_v<T, SeqStmt>) {
480  // Dynamic type-checking for a SeqStmt that could be
481  // flattened.
482  if (auto* op = stmt_or_seq.template as<SeqStmtNode>()) {
483  operator()(0, op->seq);
484  return;
485  }
486  }
487 
488  if constexpr (std::is_base_of_v<T, Evaluate>) {
489  // Evaluate(0) is used to represent a no-op, and may be
490  // generated by previous calls to SeqStmt::Flatten(). These
491  // should be removed to ensure that Flatten(a+b) is equivalent
492  // to Flatten(Flatten(a), Flatten(b)).
493  if (auto* op = stmt_or_seq.template as<EvaluateNode>()) {
494  if (auto* as_int = op->value.template as<IntImmNode>(); as_int && as_int->value == 0) {
495  return;
496  }
497  }
498  }
499 
500  if constexpr (std::is_base_of_v<Stmt, T>) {
501  // Any other Stmt type just gets appended.
502  seq_->push_back(stmt_or_seq);
503  } else {
504  // Anything else is treated as an iterable of Stmt.
505  for (auto v : stmt_or_seq) {
506  this->operator()(0, v);
507  }
508  }
509  }
510 
511  private:
512  ffi::Array<Stmt>* seq_;
513  };
