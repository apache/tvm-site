438  ffi::String storage_scope = "global", bool is_size_var = false,
439  bool is_unknown_type = false) {
440  Type type_annotation{nullptr};
441  if (is_unknown_type && storage_scope == "global") {
442  type_annotation = PrimType(runtime::DataType::Handle());
443  } else {
444  type_annotation = PointerType(PrimType(dtype), storage_scope);
445  }
446  return is_size_var ? tvm::tirx::SizeVar("", type_annotation)
447  : tvm::tirx::Var("", type_annotation);
