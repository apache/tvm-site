38 TVM_ALWAYS_INLINE ffi::Any CallPackedWithArgsInArray(const ffi::Function f,
39  const Array<Any>& args) {
40  size_t num_args = args.size();
41  std::vector<AnyView> packed_args(num_args);
42  for (size_t i = 0; i < num_args; ++i) {
43  packed_args[i] = args[i];
44  }
45  Any rv;
46  f.CallPacked(ffi::PackedArgs(packed_args.data(), packed_args.size()), &rv);
47  return rv;
48 }

63  void VisitAttrs(tvm::AttrVisitor* v) {
64  v->Visit("transform_func_key", &transform_func_key);
65  v->Visit("transform_func_args", &transform_func_args);
66  v->Visit("constr_func_key", &constr_func_key);
67  v->Visit("constr_func_args", &constr_func_args);
68  }
69 
71  const ffi::Function GetConstrFunc() {
72  const auto constr_func = tvm::ffi::Function::GetGlobal(constr_func_key);
73  ICHECK(constr_func.has_value()) << "constr_func_key is not registered: " << constr_func_key;
74  return *std::move(constr_func);
75  }
76 
78  const ffi::Function GetTransformFunc() {
79  auto transform_func = tvm::ffi::Function::GetGlobal(transform_func_key);
80  ICHECK(transform_func.has_value())
81  << "transform_func_key is not registered: " << transform_func_key;
82  return *std::move(transform_func);
83  }
84 
86  bool CheckConstr(IRModule mod) {
87  Array<Any> args(constr_func_args);
88  args.insert(args.begin(), GetRef<IRModule>(mod.CopyOnWrite()));
89  return CallPackedWithArgsInArray(GetConstrFunc(), args).cast<bool>();
90  }
91 
93  IRModule ApplyTransformFunc(IRModule mod) {
94  // Apply transformation when constraint is satisfied.
95  if (CheckConstr(mod)) {
96  Array<Any> args(transform_func_args);
97  args.insert(args.begin(), GetRef<IRModule>(mod.CopyOnWrite()));
98  return CallPackedWithArgsInArray(GetTransformFunc(), args).cast<IRModule>();
99  }
100  return mod;
101  }
102 
107  ObjectRef AsJSON() const;
108 
109  static constexpr const char* _type_key = "relax.tuning_api.Choice";
110  TVM_DECLARE_BASE_OBJECT_INFO(ChoiceNode, Object);
111 };

170 class TraceNode : public runtime::Object {
171  public:
173  IRModule in_mod;
175  mutable IRModule out_mod;
176  // TODO(sunggg): can we move knobs and decisions into private?
178  Array<Knob> knobs;
180  Array<String> decisions;
182  mutable double perf = -1;
184  mutable int size = 0;
186  virtual ~TraceNode() = default;
187 
188  void VisitAttrs(tvm::AttrVisitor* v) {
189  v->Visit("in_mod", &in_mod);
190  v->Visit("out_mod", &out_mod);
191  v->Visit("knobs", &knobs);
192  v->Visit("decisions", &decisions);
193  v->Visit("perf", &perf);
194  v->Visit("size", &size);
195  }
196 
198  bool Verify() const {
199  if (knobs.size() != decisions.size()) return false;
200  int n = knobs.size();
201  for (int i = 0; i < n; i++) {
202  if (!knobs[i]->IsValidDecision(decisions[i])) return false;
203  }
204  return true;
205  }
206 
208  IRModule Add(Knob knob, String decision) {
209  out_mod = knob->Apply(out_mod, decision);
210  knobs.push_back(knob);
211  decisions.push_back(decision);
212  // perf number should be initialized after new decision is applied.
213  perf = -1;
214  // increment history size.
215  size++;
216  return out_mod;
217  }
218 
224  ObjectRef AsJSON(bool include_in_mod = true) const;
225 
227  void SetPerf(double _perf) { perf = _perf; }
229  void SetOutMod(IRModule mod_) { out_mod = mod_; }
