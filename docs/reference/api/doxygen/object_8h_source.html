873 template <typename TargetType>
874 inline bool Object::IsInstance() const {
875  const Object* self = this;
876  // NOTE: the following code can be optimized by
877  // compiler dead-code elimination for already known constants.
878  if (self != nullptr) {
879  // Everything is a subclass of object.
880  if (std::is_same<TargetType, Object>::value) return true;
881  if (TargetType::_type_final) {
882  // if the target type is a final type
883  // then we only need to check the equivalence.
884  return self->type_index_ == TargetType::RuntimeTypeIndex();
885  } else {
886  // if target type is a non-leaf type
887  // Check if type index falls into the range of reserved slots.
888  uint32_t begin = TargetType::RuntimeTypeIndex();
889  // The condition will be optimized by constant-folding.
890  if (TargetType::_type_child_slots != 0) {
891  uint32_t end = begin + TargetType::_type_child_slots;
892  if (self->type_index_ >= begin && self->type_index_ < end) return true;
893  } else {
894  if (self->type_index_ == begin) return true;
895  }
896  if (!TargetType::_type_child_slots_can_overflow) return false;
897  // Invariance: parent index is always smaller than the child.
898  if (self->type_index_ < TargetType::RuntimeTypeIndex()) return false;
899  // The rare slower-path, check type hierarchy.
900  return self->DerivedFrom(TargetType::RuntimeTypeIndex());
901  }
902  } else {
903  return false;
904  }
905 }
