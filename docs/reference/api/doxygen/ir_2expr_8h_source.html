652 /* \brief Allow tvm.GLobalVar as key in STL tables
653  *
654  * For most IR expressions, it would be ambiguous whether the
655  * expression should follow reference equality or structural equality.
656  * This is not the case for variables, which do not contain nested
657  * internal structure, and are frequently used as keys in lookup
658  * tables.
659  *
660  * Providing `std::hash` and `std::equal_to` specializations for
661  * `tvm::GlobalVar` allows it to be used as a key in STL tables. For
662  * other IR expressions, the user must specify the type of equality
663  * used (e.g. `std::unordered_set<T, StructuralHash, StructuralEqual>`
664  * or `std::unordered_set<T, ffi::ObjectPtrHash, ffi::ObjectPtrEqual>`).
665  */
666 template <>
667 struct std::hash<tvm::GlobalVar> {
668  std::size_t operator()(const tvm::GlobalVar& var) const { return tvm::ffi::ObjectPtrHash()(var); }
669 };
670 
671 template <>
672 struct std::equal_to<tvm::GlobalVar> {
673  bool operator()(const tvm::GlobalVar& var_a, const tvm::GlobalVar& var_b) const {
674  return tvm::ffi::ObjectPtrEqual()(var_a, var_b);
675  }
676 };
