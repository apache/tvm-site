785 /* \brief Allow tvm.GLobalVar as key in STL tables
786  *
787  * For most IR expressions, it would be ambiguous whether the
788  * expression should follow reference equality or structural equality.
789  * This is not the case for variables, which do not contain nested
790  * internal structure, and are frequently used as keys in lookup
791  * tables.
792  *
793  * Providing `std::hash` and `std::equal_to` specializations for
794  * `tvm::GlobalVar` allows it to be used as a key in STL tables. For
795  * other IR expressions, the user must specify the type of equality
796  * used (e.g. `std::unordered_set<T, StructuralHash, StructuralEqual>`
797  * or `std::unordered_set<T, ObjectPtrHash, ObjectPtrEqual>`).
798  */
799 template <>
800 struct std::hash<tvm::GlobalVar> {
801  std::size_t operator()(const tvm::GlobalVar& var) const {
802  return tvm::runtime::ObjectPtrHash()(var);
803  }
804 };
805 
806 template <>
807 struct std::equal_to<tvm::GlobalVar> {
808  bool operator()(const tvm::GlobalVar& var_a, const tvm::GlobalVar& var_b) const {
809  return tvm::runtime::ObjectPtrEqual()(var_a, var_b);
810  }
811 };
