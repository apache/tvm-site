772 // common rule for RetValue and ArgValue
773 template <>
774 struct PackedFuncValueConverter<PrimExpr> {
775  static PrimExpr From(const TVMPODValue_& val) {
776  if (val.type_code() == kTVMNullptr) {
777  return PrimExpr(ObjectPtr<Object>(nullptr));
778  }
779  if (val.type_code() == kDLInt) {
780  int64_t value = val.operator int64_t();
781  if (value > std::numeric_limits<int>::max() || value < std::numeric_limits<int>::min()) {
782  return IntImm(runtime::DataType::Int(64), value);
783  }
784  return IntImm(runtime::DataType::Int(32), val.operator int());
785  }
786  if (val.type_code() == kDLFloat) {
787  return FloatImm(runtime::DataType::Float(32), val.operator double());
788  }
789 
790  return PrimExpr::FromObject_(val.AsObjectRef<ObjectRef>());
791  }
792 };

808 struct PackedFuncValueConverter<tvm::Bool> {

818  return val.AsObjectRef<tvm::Bool>();
