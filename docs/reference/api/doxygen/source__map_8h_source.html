153  bool SEqualReduce(const SequentialSpanNode* other, SEqualReducer equal) const {
154  if (spans.size() != other->spans.size()) {
155  return false;
156  }
157 
158  for (size_t i = 0, e = spans.size(); i != e; ++i) {
159  if (!StructuralEqual()(spans[i], other->spans[i])) {
160  return false;
161  }
162  }
163  return true;
