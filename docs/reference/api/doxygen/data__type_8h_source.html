44  public:
53  enum TypeCode {
54  kInt = kDLInt,
55  kUInt = kDLUInt,
56  kFloat = kDLFloat,
57  kHandle = TVMArgTypeCode::kTVMOpaqueHandle,
58  kBFloat = kDLBfloat,
59  kE4M3Float = 6U,
60  kE5M2Float = 7U,
61  kCustomBegin = 129
62  };
64  DataType() { data_ = DataType::Void(); }
69  explicit DataType(DLDataType dtype) : data_(dtype) {}
77  DataType(int code, int bits, int lanes, bool is_scalable = false) {
78  data_.code = static_cast<uint8_t>(code);
79  data_.bits = static_cast<uint8_t>(bits);
80  if (is_scalable) {
81  ICHECK(lanes > 1) << "Invalid value for vscale factor" << lanes;
82  }
83  data_.lanes = is_scalable ? static_cast<uint16_t>(-lanes) : static_cast<uint16_t>(lanes);
84  if (code == kBFloat) {
85  ICHECK_EQ(bits, 16);
86  }
87  if (code == kE4M3Float || code == kE5M2Float) {
88  ICHECK_EQ(bits, 8);
89  }
90  }
92  int code() const { return static_cast<int>(data_.code); }
94  int bits() const { return static_cast<int>(data_.bits); }
96  int bytes() const { return (bits() + 7) / 8; }
98  int lanes() const {
99  int lanes_as_int = static_cast<int16_t>(data_.lanes);
100  if (lanes_as_int < 0) {
101  LOG(FATAL) << "Can't fetch the lanes of a scalable vector at a compile time.";
102  }
103  return lanes_as_int;
104  }
106  int vscale_factor() const {
107  int lanes_as_int = static_cast<int16_t>(data_.lanes);
108  if (lanes_as_int >= -1) {
109  LOG(FATAL) << "A fixed length vector doesn't have a vscale factor.";
110  }
111  return -lanes_as_int;
112  }
114  int get_lanes_or_vscale_factor() { return is_scalable_vector() ? vscale_factor() : lanes(); }
116  bool is_scalar() const { return !is_scalable_vector() && lanes() == 1; }
118  bool is_bool() const { return code() == DataType::kUInt && bits() == 1; }
120  bool is_float() const { return code() == DataType::kFloat; }
122  bool is_float8() const {
123  return (code() == DataType::kFloat || code() == DataType::kE4M3Float ||
124  code() == DataType::kE5M2Float) &&
125  bits() == 8;
126  }
128  bool is_float16() const { return is_float() && bits() == 16; }
130  bool is_bfloat16() const { return code() == DataType::kBFloat && bits() == 16; }
132  bool is_int() const { return code() == DataType::kInt; }
134  bool is_uint() const { return code() == DataType::kUInt; }
136  bool is_handle() const { return code() == DataType::kHandle && !is_void(); }
138  bool is_scalable_or_fixed_length_vector() const {
139  int encoded_lanes = static_cast<int16_t>(data_.lanes);
140  return (encoded_lanes < -1) || (1 < encoded_lanes);
141  }
143  bool is_fixed_length_vector() const { return static_cast<int16_t>(data_.lanes) > 1; }
145  bool is_scalable_vector() const { return static_cast<int16_t>(data_.lanes) < -1; }
147  bool is_vector_bool() const { return is_scalable_or_fixed_length_vector() && bits() == 1; }
149  bool is_void() const { return code() == DataType::kHandle && bits() == 0 && lanes() == 0; }
155  DataType with_lanes(int lanes) const { return DataType(data_.code, data_.bits, lanes); }
161  DataType with_scalable_vscale_factor(int vscale_factor) const {
162  return DataType(data_.code, data_.bits, -vscale_factor);
163  }
169  DataType with_bits(int bits) const { return DataType(data_.code, bits, data_.lanes); }
174  DataType element_of() const { return with_lanes(1); }
178  DataType& operator=(const DataType& rhs) {
179  if (this == &rhs) {
180  return *this;
181  }
182  data_ = rhs.data_;
183  return *this;
184  }
190  bool operator==(const DataType& other) const {
191  return data_.code == other.data_.code && data_.bits == other.data_.bits &&
192  data_.lanes == other.data_.lanes;
193  }
199  bool operator!=(const DataType& other) const { return !operator==(other); }
204  operator DLDataType() const { return data_; }

291 inline int GetVectorBytes(DataType dtype) {
292  int data_bits = dtype.bits() * dtype.lanes();
293  // allow bool to exist
294  if (dtype == DataType::Bool() || dtype == DataType::Int(4) || dtype == DataType::UInt(4) ||
295  dtype == DataType::Int(1)) {
296  return 1;
297  }
298  ICHECK_EQ(data_bits % 8, 0U) << "Need to load/store by multiple of bytes";
299  return data_bits / 8;

387 inline std::ostream& operator<<(std::ostream& os, DLDataType t) { // NOLINT(*)
388  if (t.bits == 1 && t.lanes == 1 && t.code == kDLUInt) {
389  os << "bool";
390  return os;
391  }
392  if (DataType(t).is_void()) {
393  return os << "void";
394  }
395  if (t.code < DataType::kCustomBegin) {
396  os << DLDataTypeCode2Str(static_cast<DLDataTypeCode>(t.code));
397  } else {
398  os << "custom[" << GetCustomTypeName(t.code) << "]";
399  }
400  if (t.code == kTVMOpaqueHandle) return os;
401  int16_t lanes = static_cast<int16_t>(t.lanes);
402  os << static_cast<int>(t.bits);
403  if (lanes > 1) {
404  os << 'x' << lanes;
405  } else if (lanes < -1) {
406  os << "xvscalex" << -lanes;
407  }
408  return os;
409 }

422 inline DLDataType String2DLDataType(std::string s) {
423  DLDataType t;
424  // handle void type
425  if (s.length() == 0 || s == "void") {
426  t = DataType::Void();
427  return t;
428  }
429  t.bits = 32;
430  t.lanes = 1;
431  const char* scan;
432  if (s.substr(0, 3) == "int") {
433  t.code = kDLInt;
434  scan = s.c_str() + 3;
435  } else if (s.substr(0, 4) == "uint") {
436  t.code = kDLUInt;
437  scan = s.c_str() + 4;
438  } else if (s.substr(0, 5) == "float") {
439  t.code = kDLFloat;
440  scan = s.c_str() + 5;
441  } else if (s.substr(0, 6) == "handle") {
442  t.code = kTVMOpaqueHandle;
443  t.bits = 64; // handle uses 64 bit by default.
444  scan = s.c_str() + 6;
445  } else if (s == "bool") {
446  t.code = kDLUInt;
447  t.bits = 1;
448  t.lanes = 1;
449  return t;
450  } else if (s.substr(0, 6) == "bfloat") {
451  t.code = DataType::kBFloat;
452  t.bits = 16;
453  scan = s.c_str() + 6;
454  } else if (s.substr(0, 10) == "e4m3_float") {
455  t.code = DataType::kE4M3Float;
456  t.bits = 8;
457  scan = s.c_str() + 10;
458  } else if (s.substr(0, 10) == "e5m2_float") {
459  t.code = DataType::kE5M2Float;
460  t.bits = 8;
461  scan = s.c_str() + 10;
462  } else if (s.substr(0, 6) == "custom") {
463  t.code = ParseCustomDatatype(s, &scan);
464  } else {
465  scan = s.c_str();
466  LOG(FATAL) << "unknown type " << s;
467  }
468  char* xdelim; // emulate sscanf("%ux%u", bits, lanes)
469  uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
470  if (bits != 0) t.bits = bits;
471  int scalable_multiplier = 1;
472  if (strncmp(xdelim, "xvscale", 7) == 0) {
473  scalable_multiplier = -1;
474  xdelim += 7;
475  }
476  char* endpt = xdelim;
477  if (*xdelim == 'x') {
478  t.lanes = static_cast<uint16_t>(scalable_multiplier * strtoul(xdelim + 1, &endpt, 10));
479  }
480  ICHECK(endpt == s.c_str() + s.length()) << "unknown type " << s;
481  return t;
482 }
