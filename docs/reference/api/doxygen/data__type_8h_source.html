90  DataType(int code, int bits, int lanes, bool is_scalable = false) {
91  data_.code = static_cast<uint8_t>(code);
92  data_.bits = static_cast<uint8_t>(bits);
93  if (is_scalable) {
94  ICHECK(lanes > 1) << "Invalid value for vscale factor" << lanes;
95  }
96  data_.lanes = is_scalable ? static_cast<uint16_t>(-lanes) : static_cast<uint16_t>(lanes);
97  if (code == kBFloat) {
98  ICHECK_EQ(bits, 16);
99  }
100  if (code == kFloat8_e4m3fn || code == kFloat8_e5m2) {
101  ICHECK_EQ(bits, 8);
102  }
103  if (code == kFloat4_e2m1fn) {
104  ICHECK_EQ(bits, 4);
105  }
106  }
108  int code() const { return static_cast<int>(data_.code); }
110  int bits() const { return static_cast<int>(data_.bits); }
112  int bytes() const { return (bits() + 7) / 8; }
114  int lanes() const {
115  int lanes_as_int = static_cast<int16_t>(data_.lanes);
116  if (lanes_as_int < 0) {
117  LOG(FATAL) << "Can't fetch the lanes of a scalable vector at a compile time.";
118  }
119  return lanes_as_int;
120  }
122  int vscale_factor() const {
123  int lanes_as_int = static_cast<int16_t>(data_.lanes);
124  if (lanes_as_int >= -1) {
125  LOG(FATAL) << "A fixed length vector doesn't have a vscale factor.";
126  }
127  return -lanes_as_int;
128  }
130  int get_lanes_or_vscale_factor() const {
131  return is_scalable_vector() ? vscale_factor() : lanes();
132  }
134  bool is_scalar() const { return !is_scalable_vector() && lanes() == 1; }

324 inline int GetVectorBytes(DataType dtype) {
325  int data_bits = dtype.bits() * dtype.lanes();
326  // allow bool to exist
327  if (dtype == DataType::Bool() || dtype == DataType::Int(4) || dtype == DataType::UInt(4) ||
328  dtype == DataType::Int(1) || dtype == DataType::NVFloat4E2M1FN()) {
329  return 1;
330  }
331  ICHECK_EQ(data_bits % 8, 0U) << "Need to load/store by multiple of bytes";
332  return data_bits / 8;
333 }
